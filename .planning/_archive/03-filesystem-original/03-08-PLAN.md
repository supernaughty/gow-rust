    //! Recursive delete using walkdir.contents_first(true) per Pitfall 6.
    //!
    //! We MUST use contents_first(true). Without it, the iterator yields the
    //! parent dir before its children, and remove_dir fails with ENOTEMPTY.

    use std::path::Path;

    use walkdir::WalkDir;

    use crate::RmOpts;

    pub fn remove_tree(root: &Path, opts: &RmOpts) -> i32 {
        let mut exit_code = 0;
        for entry in WalkDir::new(root)
            .contents_first(true)
            .follow_links(false)
            .into_iter()
        {
            let entry = match entry {
                Ok(e) => e,
                Err(err) => {
                    if !opts.force {
                        eprintln!(
                            "rm: cannot access '{}': {}",
                            err.path().map(|p| p.display().to_string()).unwrap_or_default(),
                            err.io_error().map(|e| e.to_string()).unwrap_or_else(|| err.to_string())
                        );
                    }
                    exit_code = 1;
                    continue;
                }
            };

            let path = entry.path();
            let ft = entry.file_type();

            if ft.is_dir() {
                // Try remove_dir — contents should be gone by now (contents_first).
                if let Err(e) = std::fs::remove_dir(path) {
                    if !opts.force {
                        eprintln!("rm: cannot remove '{}': {}", path.display(), e);
                    }
                    exit_code = 1;
                } else if opts.verbose {
                    println!("removed directory '{}'", path.display());
                }
            } else {
                // File / symlink entry: clear RO if -f, then remove_file.
                if opts.force {
                    if let Ok(md) = std::fs::symlink_metadata(path) {
                        if gow_core::fs::is_readonly(&md) {
                            let _ = gow_core::fs::clear_readonly(path);
                        }
                    }
                }
                if let Err(e) = std::fs::remove_file(path) {
                    if !opts.force {
                        eprintln!("rm: cannot remove '{}': {}", path.display(), e);
                    }
                    exit_code = 1;
                } else if opts.verbose {
                    println!("removed '{}'", path.display());
                }
            }
        }
        exit_code
    }

    //! `uu_rm`: GNU `rm` — remove files and directories (FILE-05).
    //!
    //! Safety gates per D-42: refuses to operate recursively on Windows drive
    //! roots (C:\, D:\, \\server\share) or POSIX `/` unless --no-preserve-root.
    //! Per D-45, reads stdin for prompts only when it is a tty.

    mod recurse;

    use std::ffi::OsString;
    use std::io::{self, BufRead, IsTerminal, Write};
    use std::path::Path;

    use clap::{Arg, ArgAction, Command};

    #[derive(Debug, Clone)]
    pub struct RmOpts {
        pub recursive: bool,
        pub force: bool,
        pub interactive: bool,
        pub verbose: bool,
        pub preserve_root: bool,
    }

    fn uu_app() -> Command {
        Command::new("rm")
            .about("GNU rm — remove files and directories")
            .arg(Arg::new("recursive").short('r').short_alias('R').long("recursive").action(ArgAction::SetTrue))
            .arg(Arg::new("force").short('f').long("force").action(ArgAction::SetTrue))
            .arg(Arg::new("interactive").short('i').long("interactive").action(ArgAction::SetTrue))
            .arg(Arg::new("verbose").short('v').long("verbose").action(ArgAction::SetTrue))
            .arg(Arg::new("preserve-root").long("preserve-root").action(ArgAction::SetTrue)
                .help("don't remove / (default)"))
            .arg(Arg::new("no-preserve-root").long("no-preserve-root").action(ArgAction::SetTrue)
                .help("do not treat / specially"))
            .arg(Arg::new("operands").action(ArgAction::Append).trailing_var_arg(true))
    }

    pub fn uumain<I: IntoIterator<Item = OsString>>(args: I) -> i32 {
        gow_core::init();

        let matches = gow_core::args::parse_gnu(uu_app(), args);

        let opts = RmOpts {
            recursive: matches.get_flag("recursive"),
            force: matches.get_flag("force"),
            interactive: matches.get_flag("interactive"),
            verbose: matches.get_flag("verbose"),
            preserve_root: !matches.get_flag("no-preserve-root"),
        };

        let operands: Vec<String> = matches
            .get_many::<String>("operands")
            .map(|iter| iter.cloned().collect())
            .unwrap_or_default();

        if operands.is_empty() {
            if !opts.force {
                eprintln!("rm: missing operand");
                return 1;
            }
            return 0;
        }

        let mut exit_code = 0;
        for op in &operands {
            let converted = gow_core::path::try_convert_msys_path(op);
            let path = Path::new(&converted);

            // D-42 safety gate: refuse drive roots / POSIX root for -r unless override.
            if opts.recursive && opts.preserve_root
                && (gow_core::fs::is_drive_root(path) || is_posix_root(path))
            {
                eprintln!(
                    "rm: it is dangerous to operate recursively on '{converted}'\nrm: use --no-preserve-root to override this failsafe"
                );
                exit_code = 1;
                continue;
            }

            // Check existence.
            let md = match std::fs::symlink_metadata(path) {
                Ok(m) => m,
                Err(e) if e.kind() == io::ErrorKind::NotFound => {
                    if opts.force { continue; }
                    eprintln!("rm: cannot remove '{converted}': No such file or directory");
                    exit_code = 1;
                    continue;
                }
                Err(e) => {
                    if !opts.force {
                        eprintln!("rm: cannot remove '{converted}': {e}");
                    }
                    exit_code = 1;
                    continue;
                }
            };

            // Directory without -r → error.
            if md.is_dir() && !opts.recursive {
                eprintln!("rm: cannot remove '{converted}': Is a directory");
                exit_code = 1;
                continue;
            }

            // Interactive prompt (D-43).
            if opts.interactive {
                let prompt = if md.is_dir() {
                    format!("rm: descend into directory '{converted}'? ")
                } else {
                    format!("rm: remove regular file '{converted}'? ")
                };
                if !prompt_user(&prompt) {
                    continue;
                }
            }

            if md.is_dir() {
                let rc = recurse::remove_tree(path, &opts);
                if rc != 0 { exit_code = rc; }
            } else {
                if let Err(e) = delete_one_file(path, &opts) {
                    if !opts.force {
                        eprintln!("rm: cannot remove '{converted}': {e}");
                    }
                    exit_code = 1;
                }
            }
        }
        exit_code
    }

    fn delete_one_file(path: &Path, opts: &RmOpts) -> io::Result<()> {
        // D-45: read-only handling.
        let md = std::fs::symlink_metadata(path)?;
        if gow_core::fs::is_readonly(&md) {
            if !opts.force {
                // Only prompt if stdin is a tty (D-45).
                if io::stdin().is_terminal() {
                    let prompt = format!("rm: remove write-protected regular file '{}'? ", path.display());
                    if !prompt_user(&prompt) {
                        return Ok(());
                    }
                } else {
                    return Err(io::Error::new(
                        io::ErrorKind::PermissionDenied,
                        "Permission denied",
                    ));
                }
            }
            gow_core::fs::clear_readonly(path)?;
        }
        std::fs::remove_file(path)?;
        if opts.verbose {
            println!("removed '{}'", path.display());
        }
        Ok(())
    }

    fn is_posix_root(path: &Path) -> bool {
        let s = path.to_string_lossy();
        s == "/" || s == "\\"
    }

    fn prompt_user(prompt: &str) -> bool {
        let mut stderr = io::stderr();
        let _ = write!(stderr, "{prompt}");
        let _ = stderr.flush();
        let mut line = String::new();
        if io::stdin().lock().read_line(&mut line).is_err() {
            return false;
        }
        matches!(line.trim().chars().next(), Some('y' | 'Y'))
    }

    #[cfg(test)]
    mod tests {
        use super::*;

        #[test]
        fn is_posix_root_detection() {
            assert!(is_posix_root(Path::new("/")));
            assert!(!is_posix_root(Path::new("/etc")));
        }
    }
