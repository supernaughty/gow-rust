<!-- GNU mkdir / rmdir flags -->

| Flag | mkdir | rmdir |
|---|---|---|
| `-p`, `--parents` | create missing parents; success if already exists | remove each parent dir if it becomes empty |
| `-m MODE`, `--mode=MODE` | set file mode (octal) | n/a |
| `-v`, `--verbose` | print each dir created | print each dir removed |

<!-- Windows-specific behavior -->
- `std::fs::create_dir_all(path)` on Windows: creates the chain; if the final path already exists AS A DIRECTORY, returns Ok(()). If it exists as a FILE, returns Err(AlreadyExists).
- `std::fs::remove_dir(path)` on Windows: calls RemoveDirectoryW; fails with `ErrorKind::DirectoryNotEmpty` (code 145) if the dir has contents. On Rust 1.85 this `ErrorKind` is stable.
- `ErrorKind::DirectoryNotEmpty` added in Rust 1.83; we require 1.85, so this enum variant is available.

<!-- Q5 manual rmdir -p loop (RESEARCH.md lines 432-462) — verbatim skeleton -->
```rust
fn rmdir_parents(path: &Path) -> Result<(), GowError> {
    std::fs::remove_dir(path).map_err(|e| GowError::io("remove dir", path, e))?;
    let mut current = path.parent();
    while let Some(p) = current {
        if p.as_os_str().is_empty() { break; }
        match std::fs::remove_dir(p) {
            Ok(()) => {}
            Err(e) if is_not_empty(&e) => break,
            Err(e) => return Err(GowError::io("remove dir", p, e)),
        }
        current = p.parent();
    }
    Ok(())
}
```

```rust
//! `uu_mkdir`: GNU `mkdir` — thin `std::fs::create_dir` / `create_dir_all` wrapper.
//! MSYS pre-convert on each operand (D-26). `-p` delegates to create_dir_all (D-27).

use std::ffi::OsString;
use std::path::Path;

use clap::{Arg, ArgAction, Command};

pub fn uumain<I: IntoIterator<Item = OsString>>(args: I) -> i32 {
    gow_core::init();

    let matches = gow_core::args::parse_gnu(uu_app(), args);
    let parents = matches.get_flag("parents");
    let verbose = matches.get_flag("verbose");

    let operands: Vec<String> = matches
        .get_many::<String>("operands")
        .map(|iter| iter.cloned().collect())
        .unwrap_or_default();

    if operands.is_empty() {
        eprintln!("mkdir: missing operand");
        return 1;
    }

    let mut exit_code = 0;
    for op in &operands {
        let converted = gow_core::path::try_convert_msys_path(op);
        let path = Path::new(&converted);

        let result = if parents {
            // D-27 / Q5: std::fs::create_dir_all is POSIX-correct — returns Ok if the
            // path already exists as a directory.
            std::fs::create_dir_all(path)
        } else {
            std::fs::create_dir(path)
        };

        match result {
            Ok(()) => {
                if verbose {
                    println!("mkdir: created directory '{converted}'");
                }
            }
            Err(e) => {
                eprintln!("mkdir: cannot create directory '{converted}': {e}");
                exit_code = 1;
            }
        }
    }

    exit_code
}

fn uu_app() -> Command {
    Command::new("mkdir")
        .arg(
            Arg::new("parents")
                .short('p')
                .long("parents")
                .action(ArgAction::SetTrue)
                .help("no error if existing, make parent directories as needed"),
        )
        .arg(
            Arg::new("verbose")
                .short('v')
                .long("verbose")
                .action(ArgAction::SetTrue)
                .help("print a message for each created directory"),
        )
        .arg(
            Arg::new("operands")
                .action(ArgAction::Append)
                .trailing_var_arg(true),
        )
}
```

**Note on `-m MODE`:** Not implemented in this plan. Windows has no POSIX mode bits (file ACLs are orthogonal). GNU manual says `-m` MAY be a no-op on systems without modes. We skip adding clap's `-m` arg entirely so unknown-flag behavior remains GNU-conformant; `v2` backlog. This is consistent with CONTEXT.md Deferred Ideas pattern (no Phase 2 decision requires `-m`). If the checker flags this as a scope reduction, add `-m` as a recognized-but-ignored flag rather than dropping — but do NOT implement actual permission changes.

```rust
//! `uu_rmdir`: GNU `rmdir` with `-p` parent walk (D-28).
//! Loop stops on first non-empty parent per POSIX semantics (RESEARCH.md Q5).

use std::ffi::OsString;
use std::io;
use std::path::Path;

use clap::{Arg, ArgAction, Command};

pub fn uumain<I: IntoIterator<Item = OsString>>(args: I) -> i32 {
    gow_core::init();

    let matches = gow_core::args::parse_gnu(uu_app(), args);
    let parents = matches.get_flag("parents");
    let verbose = matches.get_flag("verbose");

    let operands: Vec<String> = matches
        .get_many::<String>("operands")
        .map(|iter| iter.cloned().collect())
        .unwrap_or_default();

    if operands.is_empty() {
        eprintln!("rmdir: missing operand");
        return 1;
    }

    let mut exit_code = 0;
    for op in &operands {
        let converted = gow_core::path::try_convert_msys_path(op);
        let path = Path::new(&converted);

        let result = if parents {
            rmdir_parents(path, verbose)
        } else {
            remove_one(path, verbose)
        };

        if let Err(e) = result {
            eprintln!("rmdir: failed to remove '{converted}': {e}");
            exit_code = 1;
        }
    }

    exit_code
}

fn remove_one(path: &Path, verbose: bool) -> io::Result<()> {
    std::fs::remove_dir(path)?;
    if verbose {
        println!("rmdir: removing directory, '{}'", path.display());
    }
    Ok(())
}

fn rmdir_parents(path: &Path, verbose: bool) -> io::Result<()> {
    // Remove the leaf first.
    remove_one(path, verbose)?;
    // Walk up, removing parents until we hit a non-empty one.
    let mut current = path.parent();
    while let Some(p) = current {
        if p.as_os_str().is_empty() {
            break;
        }
        match std::fs::remove_dir(p) {
            Ok(()) => {
                if verbose {
                    println!("rmdir: removing directory, '{}'", p.display());
                }
            }
            Err(e) if is_not_empty(&e) => break,
            Err(e) => return Err(e),
        }
        current = p.parent();
    }
    Ok(())
}

/// Detect the platform-specific "directory not empty" error.
/// Rust 1.85 exposes `ErrorKind::DirectoryNotEmpty` on Windows; we also check the raw
/// OS code 145 as a defense-in-depth (RESEARCH.md Q5 Windows edge).
fn is_not_empty(e: &io::Error) -> bool {
    e.kind() == io::ErrorKind::DirectoryNotEmpty
        || {
            #[cfg(windows)]
            { e.raw_os_error() == Some(145) }
            #[cfg(not(windows))]
            { e.raw_os_error() == Some(39) }  // ENOTEMPTY on Linux/macOS
        }
}

fn uu_app() -> Command {
    Command::new("rmdir")
        .arg(
            Arg::new("parents")
                .short('p')
                .long("parents")
                .action(ArgAction::SetTrue)
                .help("remove DIRECTORY and its ancestors; e.g., 'rmdir -p a/b/c' is equivalent to 'rmdir a/b/c a/b a'"),
        )
        .arg(
            Arg::new("verbose")
                .short('v')
                .long("verbose")
                .action(ArgAction::SetTrue)
                .help("output a diagnostic for every directory processed"),
        )
        .arg(
            Arg::new("operands")
                .action(ArgAction::Append)
                .trailing_var_arg(true),
        )
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn is_not_empty_detects_rust_kind() {
        let e = io::Error::from(io::ErrorKind::DirectoryNotEmpty);
        assert!(is_not_empty(&e));
    }

    #[test]
    fn is_not_empty_rejects_not_found() {
        let e = io::Error::from(io::ErrorKind::NotFound);
        assert!(!is_not_empty(&e));
    }
}
```
