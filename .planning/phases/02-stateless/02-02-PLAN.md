```rust
//!
//! Embeds a Windows application manifest that enables:
//! - `activeCodePage = UTF-8` (WIN-01): process-wide ANSI APIs operate on UTF-8.
```

```rust
//! `uu_yes`: GNU `yes` — infinite repeat of a string (default `y`).
//! Throughput-optimized via 8 KiB prefill buffer + `write_all` loop.
//! Silences `BrokenPipe` (GNU behavior: `yes | head` exits 0, D-23 / RESEARCH.md Q4).

use std::ffi::OsString;
use std::io::{self, Write};

const BUF_SIZE: usize = 16 * 1024;  // 16 KiB; tested sweet spot per RESEARCH.md Q4

pub fn uumain<I: IntoIterator<Item = OsString>>(args: I) -> i32 {
    gow_core::init();

    // Collect args (skip argv[0] binary name).
    let mut iter = args.into_iter();
    let _argv0 = iter.next();
    let rest: Vec<OsString> = iter.collect();

    // Build the string to repeat (D-23).
    let payload: String = if rest.is_empty() {
        "y\n".to_string()
    } else {
        // Space-join user args + trailing newline.
        let joined: Vec<String> = rest
            .iter()
            .map(|os| os.to_string_lossy().into_owned())
            .collect();
        format!("{}\n", joined.join(" "))
    };

    run(payload.as_bytes())
}

fn run(bytes: &[u8]) -> i32 {
    let mut buffer = vec![0u8; BUF_SIZE];
    let to_write = prepare_buffer(bytes, &mut buffer);

    let stdout = io::stdout();
    let mut out = stdout.lock();

    loop {
        match out.write_all(to_write) {
            Ok(()) => {}
            Err(e) if e.kind() == io::ErrorKind::BrokenPipe => return 0,
            Err(e) => {
                eprintln!("yes: {e}");
                return 1;
            }
        }
    }
}

/// Fill `buffer` with repeated copies of `input` if `input` is short enough.
/// Returns the filled slice — may be the buffer (up to BUF_SIZE) or `input` itself.
/// Pattern verbatim from uutils yes.rs (RESEARCH.md Q4 lines 313-324).
pub(crate) fn prepare_buffer<'a>(input: &'a [u8], buffer: &'a mut [u8]) -> &'a [u8] {
    if input.len() < buffer.len() / 2 {
        let mut size = 0;
        while size + input.len() <= buffer.len() {
            buffer[size..size + input.len()].copy_from_slice(input);
            size += input.len();
        }
        &buffer[..size]
    } else {
        input
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn prepare_buffer_fills_short_input() {
        let input = b"ab";
        let mut buf = [0u8; 16];
        let out = prepare_buffer(input, &mut buf);
        assert_eq!(out.len(), 16);
        assert_eq!(out, b"abababababababab");
    }

    #[test]
    fn prepare_buffer_returns_input_if_long() {
        // input longer than buffer/2 → return input as-is
        let input = [1u8; 100];
        let mut buf = [0u8; 16];
        let out = prepare_buffer(&input, &mut buf);
        assert_eq!(out.as_ptr(), input.as_ptr());  // same slice, not buffer
        assert_eq!(out.len(), 100);
    }
}
```

```rust
//! Integration tests for `yes` (UTIL-07).
//! Covers: default y output, custom string, BrokenPipe-safe pipe exit, UTF-8 arg.

use assert_cmd::Command;
use predicates::prelude::*;
use std::io::Read;
use std::process::Stdio;

fn yes_cmd() -> Command {
    Command::cargo_bin("yes")
        .expect("yes binary not found — run `cargo build -p gow-yes` first")
}

#[test]
fn test_default_y_output_head() {
    // Capture the first 32 bytes of stdout. Since yes loops forever, we need to
    // kill it after reading. assert_cmd supports timeout via .timeout().
    let out = yes_cmd()
        .timeout(std::time::Duration::from_millis(500))
        .output()
        .expect("spawn yes");
    // yes prints forever — assert_cmd timeout terminates it; we sample the captured buffer.
    let stdout = out.stdout;
    assert!(stdout.len() >= 32, "expected at least 32 bytes, got {}", stdout.len());
    // First 32 bytes should all match 'y\n' pattern.
    let expected_prefix: Vec<u8> = std::iter::repeat(*b"y\n").take(16).flatten().collect();
    assert_eq!(&stdout[..32], &expected_prefix[..]);
}

#[test]
fn test_multi_arg_space_joined() {
    let out = yes_cmd()
        .args(["hello", "world"])
        .timeout(std::time::Duration::from_millis(500))
        .output()
        .expect("spawn yes");
    let stdout = String::from_utf8_lossy(&out.stdout);
    assert!(
        stdout.contains("hello world\n"),
        "expected 'hello world\\n' in output, got: {stdout:?}"
    );
}

#[test]
fn test_utf8_arg() {
    let out = yes_cmd()
        .arg("안녕")
        .timeout(std::time::Duration::from_millis(300))
        .output()
        .expect("spawn yes");
    let stdout = String::from_utf8_lossy(&out.stdout);
    assert!(
        stdout.contains("안녕\n"),
        "expected '안녕\\n' in output, got: {stdout:?}"
    );
}

#[test]
fn test_broken_pipe_exits_zero() {
    // Pipe to a reader that closes after reading a few bytes → yes gets BrokenPipe → exits 0.
    // Use std::process::Command directly so we can wire the pipe.
    let bin = assert_cmd::cargo::cargo_bin("yes");
    let mut child = std::process::Command::new(&bin)
        .stdout(Stdio::piped())
        .spawn()
        .expect("spawn yes");
    // Read 64 bytes, then drop the reader — this closes the pipe.
    let mut stdout = child.stdout.take().expect("stdout");
    let mut buf = [0u8; 64];
    let _ = stdout.read_exact(&mut buf);
    drop(stdout);
    // Wait for yes to exit. It should see BrokenPipe and return 0 (D-23).
    let status = child.wait().expect("wait");
    assert!(status.success(), "yes should exit 0 on BrokenPipe, got {status:?}");
}

#[test]
fn test_single_arg_with_newline_in_output() {
    let out = yes_cmd()
        .arg("foo")
        .timeout(std::time::Duration::from_millis(300))
        .output()
        .expect("spawn yes");
    let stdout = String::from_utf8_lossy(&out.stdout);
    // Expect 'foo\n' repeated at least twice in the output.
    let count = stdout.matches("foo\n").count();
    assert!(count >= 2, "expected at least 2 'foo\\n' occurrences, got {count} in {stdout:?}");
}
```
