<!-- tee split-writer pattern (RESEARCH.md Executive Summary line 84 + uutils Unix-only context) -->
- Open each file operand with the right mode:
  - Default: OpenOptions::new().write(true).create(true).truncate(true).open(path)
  - -a: OpenOptions::new().write(true).create(true).append(true).open(path)

- On a failed sink: report error, keep the sink OUT of the fanout list, continue with remaining
- Line buffering can be omitted — GNU tee uses block buffering by default; `-u` (unbuffered) is a POSIX extension we skip for v1

```rust
//! Platform-gated Ctrl+C suppression for `tee -i`.
//!
//! Windows: `SetConsoleCtrlHandler(None, TRUE)` causes the process to ignore
//! CTRL_C_EVENT, CTRL_BREAK_EVENT, CTRL_CLOSE_EVENT. This is the exact behavior
//! GNU `tee -i` produces on POSIX via `signal(SIGINT, SIG_IGN)`.
//!
//! Reference: RESEARCH.md Q10 lines 883-960. uutils's tee has no Windows
//! implementation of -i (noop); we are first-in-ecosystem.

#[cfg(windows)]
pub fn ignore_interrupts() -> std::io::Result<()> {
    use windows_sys::Win32::System::Console::SetConsoleCtrlHandler;
    // add=1 (TRUE), handler=None (NULL) → process ignores console ctrl events.
    let ok = unsafe { SetConsoleCtrlHandler(None, 1) };
    if ok == 0 {
        Err(std::io::Error::last_os_error())
    } else {
        Ok(())
    }
}

#[cfg(unix)]
pub fn ignore_interrupts() -> std::io::Result<()> {
    unsafe extern "C" {
        fn signal(signum: std::os::raw::c_int, handler: usize) -> usize;
    }
    const SIGINT: std::os::raw::c_int = 2;
    const SIG_IGN: usize = 1;
    let prev = unsafe { signal(SIGINT, SIG_IGN) };
    if prev == usize::MAX {
        Err(std::io::Error::last_os_error())
    } else {
        Ok(())
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn ignore_interrupts_smoke_ok() {
        // Verify the call returns Ok in a normal test environment. This doesn't prove
        // Ctrl+C is actually ignored (see RESEARCH.md Q10 "Integration 테스트 한계") —
        // just that the API succeeds when invoked from a console-attached process.
        // If the test runner is detached, the call may fail; in that case we accept Err.
        let _ = ignore_interrupts();
    }
}
```

```rust
//! `uu_tee`: GNU `tee` — split stdin to stdout + N files.
//!
//! Flags: -a/--append (append rather than truncate), -i/--ignore-interrupts
//! (Windows: SetConsoleCtrlHandler(None, TRUE); Unix: SIGINT SIG_IGN).
//! MSYS pre-convert applies to each file operand (D-26).
//!
//! Reference: RESEARCH.md Q10, CONTEXT.md D-25.

mod signals;

use std::ffi::OsString;
use std::fs::{File, OpenOptions};
use std::io::{self, Read, Write};
use std::path::Path;

use clap::{Arg, ArgAction, Command};

pub fn uumain<I: IntoIterator<Item = OsString>>(args: I) -> i32 {
    gow_core::init();

    let matches = gow_core::args::parse_gnu(uu_app(), args);
    let append = matches.get_flag("append");
    let ignore_int = matches.get_flag("ignore-interrupts");

    if ignore_int {
        // Best-effort; silent failure if stdout is not console-attached (see signals module docs).
        let _ = signals::ignore_interrupts();
    }

    let operands: Vec<String> = matches
        .get_many::<String>("operands")
        .map(|iter| iter.cloned().collect())
        .unwrap_or_default();

    // Open each output file; record both the writer and an error status.
    let mut outputs: Vec<Box<dyn Write>> = Vec::with_capacity(operands.len() + 1);

    // stdout always goes first. We lock once for the lifetime of this function.
    // Note: we deliberately DO NOT lock here because adding StdoutLock to a
    // Box<dyn Write> requires lifetime coupling. Instead, wrap in a thin struct
    // that calls stdout().lock() per-write — this is what uutils does.
    outputs.push(Box::new(io::stdout()));

    let mut file_errors = 0u32;
    for op in &operands {
        let converted = gow_core::path::try_convert_msys_path(op);
        let path = Path::new(&converted);
        match open_output_file(path, append) {
            Ok(file) => outputs.push(Box::new(file)),
            Err(e) => {
                eprintln!("tee: {converted}: {e}");
                file_errors += 1;
            }
        }
    }

    // Copy stdin to every successfully-opened output sink.
    let stdin = io::stdin();
    let mut input = stdin.lock();
    let mut buf = [0u8; 8192];

    loop {
        match input.read(&mut buf) {
            Ok(0) => break,
            Ok(n) => {
                // Fanout: write to each sink. If any sink errors, log + drop it.
                outputs.retain_mut(|sink| match sink.write_all(&buf[..n]) {
                    Ok(()) => true,
                    Err(e) => {
                        // Log error but continue. We don't know which sink this is after
                        // the Box::<dyn Write> erasure, so the error message is generic.
                        eprintln!("tee: write error: {e}");
                        false
                    }
                });
            }
            Err(e) if e.kind() == io::ErrorKind::Interrupted => continue,
            Err(e) => {
                eprintln!("tee: read error: {e}");
                return 1;
            }
        }
    }

    // Flush each sink before exit.
    for sink in outputs.iter_mut() {
        let _ = sink.flush();
    }

    if file_errors > 0 { 1 } else { 0 }
}

fn open_output_file(path: &Path, append: bool) -> io::Result<File> {
    let mut opts = OpenOptions::new();
    opts.write(true).create(true);
    if append {
        opts.append(true);
    } else {
        opts.truncate(true);
    }
    opts.open(path)
}

fn uu_app() -> Command {
    Command::new("tee")
        .arg(
            Arg::new("append")
                .short('a')
                .long("append")
                .action(ArgAction::SetTrue)
                .help("append to the given FILEs, do not overwrite"),
        )
        .arg(
            Arg::new("ignore-interrupts")
                .short('i')
                .long("ignore-interrupts")
                .action(ArgAction::SetTrue)
                .help("ignore interrupt signals"),
        )
        .arg(
            Arg::new("operands")
                .action(ArgAction::Append)
                .trailing_var_arg(true),
        )
}
```

**Note on stdout handling without explicit lock:** Wrapping `io::stdout()` (which is `Stdout`, unlocked) in `Box<dyn Write>` means each `write_all` implicitly acquires/releases the lock per call. This is slightly slower than holding `StdoutLock` for the whole loop, but avoids the lifetime gymnastics of `Box<dyn Write + 'a>`. Acceptable for `tee`, which is inherently I/O-bound. If perf matters later, switch to a bespoke enum `enum Sink { Stdout(StdoutLock<'static>), File(File) }`.
