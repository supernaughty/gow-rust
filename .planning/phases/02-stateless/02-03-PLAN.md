//! GNU `echo -e` escape-sequence state machine.
//!
//! Parses `\n`, `\t`, `\r`, `\f`, `\v`, `\a`, `\b`, `\e`, `\\`, `\0NNN` (octal, up
//! to 3 digits), `\xHH` (hex, up to 2 digits), and `\c` (early break — see Control).
//! Unknown escapes (`\z`) are preserved verbatim, matching GNU's fallback.
//!
//! Reference: RESEARCH.md Q9 lines 754-880. Unit tests in this file.

use std::io::{self, Write};

/// Controls whether the caller should append a trailing newline after `write_escaped` returns.
/// `Break` means `\c` was encountered — caller MUST NOT append the newline.
#[derive(Debug, PartialEq, Eq)]
pub enum Control {
    Continue,
    Break,
}

/// Write `bytes` to `out`, interpreting backslash escapes per GNU `echo -e`.
///
/// Returns `Control::Break` if a `\c` sequence was encountered (caller must suppress
/// any trailing newline). Otherwise returns `Control::Continue`.
pub fn write_escaped<W: Write>(bytes: &[u8], out: &mut W) -> io::Result<Control> {
    let mut i = 0;
    while i < bytes.len() {
        let b = bytes[i];
        if b != b'\\' {
            out.write_all(&[b])?;
            i += 1;
            continue;
        }
        // Starts escape.
        if i + 1 >= bytes.len() {
            // Trailing bare `\` — preserve literal.
            out.write_all(&[b'\\'])?;
            return Ok(Control::Continue);
        }
        match bytes[i + 1] {
            b'\\' => { out.write_all(&[b'\\'])?; i += 2; }
            b'a'  => { out.write_all(&[0x07])?;  i += 2; }
            b'b'  => { out.write_all(&[0x08])?;  i += 2; }
            b'c'  => return Ok(Control::Break),
            b'e'  => { out.write_all(&[0x1B])?;  i += 2; }
            b'f'  => { out.write_all(&[0x0C])?;  i += 2; }
            b'n'  => { out.write_all(&[0x0A])?;  i += 2; }
            b'r'  => { out.write_all(&[0x0D])?;  i += 2; }
            b't'  => { out.write_all(&[0x09])?;  i += 2; }
            b'v'  => { out.write_all(&[0x0B])?;  i += 2; }
            b'0'  => {
                let (byte, consumed) = parse_octal(&bytes[i + 2..]);
                out.write_all(&[byte])?;
                i += 2 + consumed;
            }
            b'x'  => {
                let (byte, consumed) = parse_hex(&bytes[i + 2..]);
                if consumed == 0 {
                    // No hex digit after \x — preserve literal "\xREST" as backslash+x
                    // and continue at the char after 'x'.
                    out.write_all(&[b'\\', b'x'])?;
                    i += 2;
                } else {
                    out.write_all(&[byte])?;
                    i += 2 + consumed;
                }
            }
            other => {
                // Unknown escape → emit backslash + char verbatim (GNU behavior).
                out.write_all(&[b'\\', other])?;
                i += 2;
            }
        }
    }
    Ok(Control::Continue)
}

/// Parse up to 3 octal digits from the start of `rest`. Returns (byte, digits_consumed).
pub fn parse_octal(rest: &[u8]) -> (u8, usize) {
    let mut val: u8 = 0;
    let mut n = 0;
    while n < 3 && n < rest.len() && (b'0'..=b'7').contains(&rest[n]) {
        val = val.wrapping_mul(8).wrapping_add(rest[n] - b'0');
        n += 1;
    }
    (val, n)
}

/// Parse up to 2 hex digits from the start of `rest`. Returns (byte, digits_consumed).
pub fn parse_hex(rest: &[u8]) -> (u8, usize) {
    let mut val: u8 = 0;
    let mut n = 0;
    while n < 2 && n < rest.len() && rest[n].is_ascii_hexdigit() {
        let digit = match rest[n] {
            b'0'..=b'9' => rest[n] - b'0',
            b'a'..=b'f' => rest[n] - b'a' + 10,
            b'A'..=b'F' => rest[n] - b'A' + 10,
            _ => unreachable!(),
        };
        val = val.wrapping_mul(16).wrapping_add(digit);
        n += 1;
    }
    (val, n)
}

```rust
//! `uu_echo`: GNU `echo` with -n / -e / -E semantics.
//!
//! Entry: `pub fn uumain`. Escape parsing is delegated to `escape::write_escaped`.
//! Reference: RESEARCH.md Q9 (state machine) and CONTEXT.md D-21.

mod escape;

use std::ffi::OsString;
use std::io::{self, Write};

use clap::{Arg, ArgAction, Command};

use crate::escape::{write_escaped, Control};

pub fn uumain<I: IntoIterator<Item = OsString>>(args: I) -> i32 {
    gow_core::init();

    let matches = gow_core::args::parse_gnu(uu_app(), args);

    let no_newline = matches.get_flag("no-newline");
    // -e and -E are mutually overriding; last-specified wins (clap's default with ArgAction::SetTrue
    // does NOT track ordering, so we approximate by: -E sets to false, -e sets to true, default is
    // -E (escapes OFF). If both flags are present, we treat the LATER one as winning — clap does not
    // preserve order, so for simplicity: if -e present → interpret escapes = true UNLESS -E also
    // present in which case disable. This matches BSD echo, not GNU. TODO: if tests require GNU's
    // last-wins ordering, switch to raw argv scan. For now, fail-safe: -E overrides -e.
    let interpret_escapes = matches.get_flag("interpret-escapes")
        && !matches.get_flag("disable-escapes");

    // Collect positional args (may be empty → echo prints just newline).
    let words: Vec<String> = matches
        .get_many::<String>("args")
        .map(|iter| iter.cloned().collect())
        .unwrap_or_default();
    let joined = words.join(" ");
    let bytes = joined.as_bytes();

    let stdout = io::stdout();
    let mut out = stdout.lock();

    let control = if interpret_escapes {
        match write_escaped(bytes, &mut out) {
            Ok(c) => c,
            Err(e) => {
                eprintln!("echo: {e}");
                return 1;
            }
        }
    } else {
        if let Err(e) = out.write_all(bytes) {
            eprintln!("echo: {e}");
            return 1;
        }
        Control::Continue
    };

    // Trailing newline suppression rules (D-21 + Q9):
    //   - `-n` suppresses newline always
    //   - `\c` escape seen (Control::Break) suppresses newline (and requires -e to reach)
    let suppress_newline = no_newline || matches!(control, Control::Break);
    if !suppress_newline {
        if let Err(e) = out.write_all(b"\n") {
            eprintln!("echo: {e}");
            return 1;
        }
    }
    0
}

fn uu_app() -> Command {
    Command::new("echo")
        .disable_help_flag(false)
        .disable_version_flag(false)
        .arg(
            Arg::new("no-newline")
                .short('n')
                .action(ArgAction::SetTrue)
                .help("do not append trailing newline"),
        )
        .arg(
            Arg::new("interpret-escapes")
                .short('e')
                .action(ArgAction::SetTrue)
                .help("interpret backslash escapes"),
        )
        .arg(
            Arg::new("disable-escapes")
                .short('E')
                .action(ArgAction::SetTrue)
                .help("disable backslash escapes (default)"),
        )
        .arg(
            Arg::new("args")
                .action(ArgAction::Append)
                .trailing_var_arg(true)
                .allow_hyphen_values(true),
        )
}
```

**Note on `-e` / `-E` precedence:** If integration tests expose a behavior mismatch with GNU's ordered-last-wins rule (e.g., `echo -e -E '\t'` should print literal `\t`, but `echo -E -e '\t'` should interpret), convert to a raw-argv scan before clap. The current logic "E overrides e" is conservative and matches the common case. This is a Claude's Discretion area per CONTEXT.md.
