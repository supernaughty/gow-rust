    ```rust
    //! Initial tail emission (non-follow path). Reads the file and emits the
    //! last N lines or last N bytes. D-48 raw-bytes.

    use std::io::{self, Read, Write};

    /// Emit the last `n` lines (terminated by b'\n') from the reader into writer.
    /// If the reader has fewer than `n` lines, emit all of them.
    ///
    /// Implementation: read all bytes, scan backwards for `n` '\n' boundaries.
    /// Acceptable for typical tail usage (log files up to hundreds of MB).
    pub fn tail_last_lines<R: Read, W: Write>(
        mut reader: R,
        writer: &mut W,
        n: u64,
    ) -> io::Result<()> {
        let mut buf = Vec::new();
        reader.read_to_end(&mut buf)?;
        if n == 0 { return Ok(()); }

        // Count newlines from the end, capturing the index of the n-th one from end.
        let mut count = 0u64;
        let mut start = 0;
        for i in (0..buf.len()).rev() {
            if buf[i] == b'\n' {
                count += 1;
                if count == n + 1 {
                    start = i + 1;
                    break;
                }
            }
        }
        writer.write_all(&buf[start..])?;
        Ok(())
    }

    /// Emit the last `n` bytes from the reader. If the reader has fewer, emit all.
    pub fn tail_last_bytes<R: Read, W: Write>(
        mut reader: R,
        writer: &mut W,
        n: u64,
    ) -> io::Result<()> {
        let mut buf = Vec::new();
        reader.read_to_end(&mut buf)?;
        let start = buf.len().saturating_sub(n as usize);
        writer.write_all(&buf[start..])?;
        Ok(())
    }

    /// Emit from line `n` onward (GNU `tail -n +N`). Counts lines from 1.
    pub fn tail_from_line<R: Read, W: Write>(
        mut reader: R,
        writer: &mut W,
        n: u64,
    ) -> io::Result<()> {
        let mut buf = Vec::new();
        reader.read_to_end(&mut buf)?;
        if n <= 1 {
            writer.write_all(&buf)?;
            return Ok(());
        }
        let mut lines_seen = 1u64;
        let mut start = 0;
        for (i, &b) in buf.iter().enumerate() {
            if b == b'\n' {
                lines_seen += 1;
                if lines_seen == n {
                    start = i + 1;
                    break;
                }
            }
        }
        writer.write_all(&buf[start..])?;
        Ok(())
    }

    #[cfg(test)]
    mod tests {
        use super::*;

        #[test]
        fn last_lines_basic() {

            let mut out = Vec::new();
            tail_last_lines(input, &mut out, 2).unwrap();
            assert_eq!(out, b"c\nd\n");
        }

        #[test]
        fn last_lines_more_than_file() {
            let input: &[u8] = b"a\nb\n";
            let mut out = Vec::new();
            tail_last_lines(input, &mut out, 10).unwrap();

        }

        #[test]

            let input: &[u8] = b"a\nb\nc";
            let mut out = Vec::new();
            tail_last_lines(input, &mut out, 2).unwrap();
            assert_eq!(out, b"b\nc");
        }

        #[test]
        fn last_bytes_basic() {
            let input: &[u8] = b"hello world";
            let mut out = Vec::new();
            tail_last_bytes(input, &mut out, 5).unwrap();
    ```

    ```rust
    //! `uu_tail`: GNU `tail` — output last N lines of a file and optionally
    //! follow for appended output (TEXT-02, ROADMAP criterion 3).
    //!
    //! Numeric shorthand (D-05): `tail -5 file` works via pre-parse.
    //! Multi-file: `==> file <==` headers per D-41.
    //! Follow (-f, -F) via notify crate (D-39), parent-dir watch (Pitfall 3).

    mod initial;
    mod follow;

    use std::ffi::OsString;
    use std::fs::File;
    use std::io;
    use std::path::Path;

    use clap::{Arg, ArgAction, Command};

    fn uu_app() -> Command {
        Command::new("tail")
            .about("GNU tail — output the last part of files")
            .arg(Arg::new("lines").short('n').long("lines").num_args(1).default_value("10")
                .allow_hyphen_values(true))
            .arg(Arg::new("bytes").short('c').long("bytes").num_args(1))
            .arg(Arg::new("follow-desc").short('f').long("follow").action(ArgAction::SetTrue)
                .help("follow file descriptor (keeps handle across rename/delete)"))
            .arg(Arg::new("follow-name").short('F').action(ArgAction::SetTrue)
                .help("follow file name (reopen on rotation)"))
            .arg(Arg::new("quiet").short('q').long("quiet").action(ArgAction::SetTrue))
            .arg(Arg::new("verbose").short('v').long("verbose").action(ArgAction::SetTrue))
            .arg(Arg::new("operands").action(ArgAction::Append).trailing_var_arg(true))
    }

    /// Rewrite `-5` / `-+5` / `-5c` into canonical `-n 5` / `-n +5` / `-c 5`.
    /// Shared pattern with head.rs — could be promoted to gow-core if we get a
    /// third user. For now, duplicated for Phase 3 shipping.
    fn expand_numeric_shorthand(args: &mut Vec<OsString>) {
        let mut i = 0;
        while i < args.len() {
            let s = args[i].to_string_lossy();
            if s == "--" { break; }
            if s.len() >= 2 && s.starts_with('-') && !s.starts_with("--") {
                let tail = &s[1..];
                let is_plus_digits = tail.starts_with('+')
                    && tail.len() > 1
                    && tail[1..].chars().all(|c| c.is_ascii_digit());
                let is_digits = tail.chars().all(|c| c.is_ascii_digit());
                if is_plus_digits || is_digits {
                    let val = tail.to_string();
                    args[i] = OsString::from("-n");
                    args.insert(i + 1, OsString::from(val));
                    i += 2;
                    continue;
                }
            }
            i += 1;
        }
    }

    #[derive(Debug, Clone, Copy)]
    enum Mode {
        LastLines(u64),
        LastBytes(u64),
        FromLine(u64),
    }

    fn parse_mode(matches: &clap::ArgMatches) -> Result<Mode, String> {
        if let Some(bytes_str) = matches.get_one::<String>("bytes") {
            let n: u64 = bytes_str.parse().map_err(|_| format!("invalid number of bytes: '{bytes_str}'"))?;
            return Ok(Mode::LastBytes(n));
        }
        let lines_str = matches.get_one::<String>("lines").expect("default set").clone();
        if let Some(rest) = lines_str.strip_prefix('+') {
            let n: u64 = rest.parse().map_err(|_| format!("invalid number: '{lines_str}'"))?;
            return Ok(Mode::FromLine(n));
        }
        let n: u64 = lines_str.parse().map_err(|_| format!("invalid number of lines: '{lines_str}'"))?;
        Ok(Mode::LastLines(n))
    }

    pub fn uumain<I: IntoIterator<Item = OsString>>(args: I) -> i32 {
        gow_core::init();

        let mut argv: Vec<OsString> = args.into_iter().collect();
        expand_numeric_shorthand(&mut argv);

        let matches = gow_core::args::parse_gnu(uu_app(), argv);

        let mode = match parse_mode(&matches) {
            Ok(m) => m,
            Err(e) => { eprintln!("tail: {e}"); return 1; }
        };

        let follow_desc = matches.get_flag("follow-desc");
        let follow_name = matches.get_flag("follow-name");
        let quiet = matches.get_flag("quiet");
        let verbose = matches.get_flag("verbose");

        let operands: Vec<String> = matches
            .get_many::<String>("operands")
            .map(|iter| iter.cloned().collect())
            .unwrap_or_default();

        let mut stdout = io::stdout().lock();
        let multi = operands.len() > 1;
        let show_headers = !quiet && (multi || verbose);

        // Stdin fallback: no operand → read stdin (no follow).
        if operands.is_empty() {
            if show_headers {
                let _ = io::Write::write_all(&mut stdout, b"==> standard input <==\n");
            }
            let res = match mode {
                Mode::LastLines(n) => initial::tail_last_lines(io::stdin().lock(), &mut stdout, n),
                Mode::LastBytes(n) => initial::tail_last_bytes(io::stdin().lock(), &mut stdout, n),
                Mode::FromLine(n)  => initial::tail_from_line(io::stdin().lock(), &mut stdout, n),
            };
            if let Err(e) = res {
                eprintln!("tail: standard input: {e}");
                return 1;
            }
            return 0;
        }

        // Emit initial content for each file.
        let mut exit_code = 0;
        let mut first = true;
        for op in &operands {
            if show_headers {
                if !first {
                    let _ = io::Write::write_all(&mut stdout, b"\n");
                }
                let hdr = format!("==> {op} <==\n");
                let _ = io::Write::write_all(&mut stdout, hdr.as_bytes());
            }
            first = false;

            let converted = gow_core::path::try_convert_msys_path(op);
            let path = Path::new(&converted);
            match File::open(path) {
                Ok(f) => {
                    let res = match mode {
                        Mode::LastLines(n) => initial::tail_last_lines(f, &mut stdout, n),
                        Mode::LastBytes(n) => initial::tail_last_bytes(f, &mut stdout, n),
                        Mode::FromLine(n)  => initial::tail_from_line(f, &mut stdout, n),
                    };
                    if let Err(e) = res {
                        eprintln!("tail: error reading '{converted}': {e}");
                        exit_code = 1;
                    }
                }
                Err(e) => {
                    eprintln!("tail: cannot open '{converted}' for reading: {e}");
                    exit_code = 1;
                }
            }
        }

        // Follow phase. For multi-file follow, we serially hand off to follow()
        // per file; a full multi-file single-event-loop is out of scope for v1.
        // Typical tail -f is single-file.
        if follow_desc || follow_name {
            for op in &operands {
                let converted = gow_core::path::try_convert_msys_path(op);
                let path = Path::new(&converted);
                let rc = if follow_name {
                    follow::follow_name(path)
                } else {
                    follow::follow_descriptor(path)
                };
                if rc != 0 { exit_code = rc; }
            }
        }

        exit_code
    }
    ```

    ```rust
    //! `tail -f` and `tail -F` follow loops using notify 8.2.
    //!
    //! Pitfall 3 / D-39: watch the PARENT directory with NonRecursive mode,
    //! filter events by filename. This is the only reliable pattern on Windows
    //! because `ReadDirectoryChangesW` operates on directory handles.
    //!
    //! Latency contract (ROADMAP criterion 3):
    //! - notify Windows backend wakeup_sem: 100ms worst case
    //! - Kernel dispatch: < 50ms typical
    //! - Total P99: ~150ms. CI-safe bound: 500ms.

    use std::io::{self, Read, Seek, SeekFrom, Write};
    use std::path::{Path, PathBuf};
    use std::sync::mpsc;

    use notify::{
        event::{EventKind, ModifyKind},
        Config, Event, RecommendedWatcher, RecursiveMode, Watcher,
    };

    /// tail -f: descriptor follow. Keeps the original file handle even when the
    /// underlying path is renamed or deleted. Emits newly appended bytes on
    /// Modify events.
    pub fn follow_descriptor(path: &Path) -> i32 {
        let file = match std::fs::File::open(path) {
            Ok(f) => f,
            Err(e) => {
                eprintln!("tail: cannot open '{}' for reading: {e}", path.display());
                return 1;
            }
        };
        let parent = path.parent().map(|p| p.to_path_buf())
            .unwrap_or_else(|| PathBuf::from("."));
        let target_name = path.file_name().map(|n| n.to_os_string());

        watch_loop(file, &parent, target_name, /*reopen_on_create=*/false, path)
    }

    /// tail -F: name follow. Reopens the file when it is removed and recreated
    /// with the same name (typical log rotator behavior).
    pub fn follow_name(path: &Path) -> i32 {
        let file = match std::fs::File::open(path) {
            Ok(f) => f,
            Err(e) => {
                eprintln!("tail: cannot open '{}' for reading: {e}", path.display());
                return 1;
            }
        };
        let parent = path.parent().map(|p| p.to_path_buf())
            .unwrap_or_else(|| PathBuf::from("."));
        let target_name = path.file_name().map(|n| n.to_os_string());

        watch_loop(file, &parent, target_name, /*reopen_on_create=*/true, path)
    }

    fn watch_loop(
        mut file: std::fs::File,
        parent: &Path,
        target_name: Option<std::ffi::OsString>,
        reopen_on_create: bool,
        target_path: &Path,
    ) -> i32 {
        // Seek to end — we only emit new data from this point.
        if let Err(e) = file.seek(SeekFrom::End(0)) {
            eprintln!("tail: {}: {e}", target_path.display());
            return 1;
        }
        let mut offset = match file.stream_position() {
            Ok(o) => o,
            Err(e) => {
                eprintln!("tail: {}: {e}", target_path.display());
                return 1;
            }
        };

        let (tx, rx) = mpsc::channel::<notify::Result<Event>>();
        let mut watcher: RecommendedWatcher = match RecommendedWatcher::new(tx, Config::default()) {
            Ok(w) => w,
            Err(e) => {
                eprintln!("tail: watcher init failed: {e}");
                return 1;
            }
        };

        if let Err(e) = watcher.watch(parent, RecursiveMode::NonRecursive) {
            eprintln!("tail: watch '{}' failed: {e}", parent.display());
            return 1;
        }

        let mut stdout = io::stdout().lock();

        loop {
            let Ok(result) = rx.recv() else { break 0 };

            let event = match result {
                Ok(e) => e,
                Err(e) => {
                    eprintln!("tail: watch error: {e}");
                    continue;
                }
            };

            // Filter: only events whose any path matches our target filename.
            let matches = match &target_name {
                Some(expected) => event.paths.iter().any(|p| {
                    p.file_name().map(|got| got == expected.as_os_str()).unwrap_or(false)
                }),
                None => true,
            };
            if !matches { continue; }

            match event.kind {
                EventKind::Modify(ModifyKind::Any) | EventKind::Modify(_) => {
                    // Either appended OR truncated — disambiguate by size.
                    let current_size = match file.metadata().map(|m| m.len()) {
                        Ok(s) => s,
                        Err(_) => continue,
                    };
                    if current_size < offset {
                        // Truncation
                        let _ = writeln!(io::stderr(), "tail: {}: file truncated", target_path.display());
                        if file.seek(SeekFrom::Start(0)).is_err() { continue; }
                        offset = 0;
                    }
                    if file.seek(SeekFrom::Start(offset)).is_err() { continue; }
                    let mut buf = Vec::new();
                    let n = file.read_to_end(&mut buf).unwrap_or(0);
                    offset += n as u64;
                    let _ = stdout.write_all(&buf);
                    let _ = stdout.flush();
                }
                EventKind::Remove(_) => {
                    if reopen_on_create {
                        // -F: the file vanished; wait for Create to reopen.
                        // (Do nothing here; Create branch handles reopen.)
                    } else {
                        // -f: keep the open handle; it's still valid.
                        let _ = writeln!(io::stderr(), "tail: {}: file has been removed", target_path.display());
                    }
                }
                EventKind::Create(_) if reopen_on_create => {
                    // Try to reopen with the same filename.
                    if let Ok(f) = std::fs::File::open(target_path) {
                        file = f;
                        offset = 0;
                        // Emit everything from the start of the new file.
                        let mut buf = Vec::new();
                        let n = file.read_to_end(&mut buf).unwrap_or(0);
                        offset += n as u64;
                        let _ = stdout.write_all(&buf);
                        let _ = stdout.flush();
                    }
                }
                _ => {}
            }
        }
    }
    ```
