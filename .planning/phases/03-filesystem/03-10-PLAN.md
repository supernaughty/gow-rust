    //! `uu_ln`: GNU `ln` — make links between files (FILE-09).
    //!
    //! Windows semantics:
    //! - Hard link (`ln`): std::fs::hard_link via gow_core::fs::create_link(Hard)
    //! - Symbolic link (`ln -s`): create_link(Symbolic) which handles D-36
    //!   junction fallback for directory targets.
    //!
    //! D-36: On the FIRST junction fallback, emit `ln: symlink privilege
    //! unavailable, created junction instead` to stderr. Subsequent junctions
    //! in the same invocation are silent.

    use std::ffi::OsString;
    use std::path::{Path, PathBuf};

    use clap::{Arg, ArgAction, Command};

    use gow_core::fs::{create_link, LinkKind, LinkOutcome};

    fn uu_app() -> Command {
        Command::new("ln")
            .about("GNU ln — make links between files")
            .arg(Arg::new("symbolic").short('s').long("symbolic").action(ArgAction::SetTrue))
            .arg(Arg::new("force").short('f').long("force").action(ArgAction::SetTrue))
            .arg(Arg::new("verbose").short('v').long("verbose").action(ArgAction::SetTrue))
            .arg(Arg::new("operands").action(ArgAction::Append).trailing_var_arg(true))
    }

    pub fn uumain<I: IntoIterator<Item = OsString>>(args: I) -> i32 {
        gow_core::init();

        let matches = gow_core::args::parse_gnu(uu_app(), args);

        let symbolic = matches.get_flag("symbolic");
        let force = matches.get_flag("force");
        let verbose = matches.get_flag("verbose");

        let operands: Vec<String> = matches
            .get_many::<String>("operands")
            .map(|iter| iter.cloned().collect())
            .unwrap_or_default();

        if operands.len() < 2 {
            eprintln!("ln: missing operand");
            eprintln!("usage: ln [OPTION]... TARGET LINK_NAME");
            eprintln!("   or: ln [OPTION]... TARGET... DIRECTORY");
            return 1;
        }

        // Last operand is either the single LINK_NAME or the DIRECTORY.
        let last_raw = operands.last().unwrap().clone();
        let targets = &operands[..operands.len() - 1];
        let last_converted = gow_core::path::try_convert_msys_path(&last_raw);
        let last = Path::new(&last_converted);
        let last_is_dir = last.is_dir();

        if targets.len() > 1 && !last_is_dir {
            eprintln!("ln: target '{last_converted}' is not a directory");
            return 1;
        }

        let mut exit_code = 0;
        let mut warned_once = false;

        for target_raw in targets {
            let target_converted = gow_core::path::try_convert_msys_path(target_raw);
            let target = Path::new(&target_converted);

            // Determine LINK destination (single-link mode vs TARGETS... DIR mode).
            let link_path: PathBuf = if last_is_dir && targets.len() >= 1 && (targets.len() > 1 || last != target) {
                last.join(target.file_name().unwrap_or_default())
            } else {
                last.to_path_buf()
            };

            // Force: remove existing.
            if force && link_path.exists() {
                let md = std::fs::symlink_metadata(&link_path);
                let rm_result = match md {
                    Ok(m) if m.is_dir() && !m.file_type().is_symlink() => {
                        // Directory (possibly junction) — remove_dir won't remove junction contents.
                        std::fs::remove_dir(&link_path)
                    }
                    _ => std::fs::remove_file(&link_path),
                };
                if let Err(e) = rm_result {
                    eprintln!("ln: failed to remove existing '{}': {e}", link_path.display());
                    exit_code = 1;
                    continue;
                }
            }

            let kind = if symbolic { LinkKind::Symbolic } else { LinkKind::Hard };

            // For hard links, verify target exists (create_link will also fail, but we want a cleaner message).
            if !symbolic && !target.exists() {
                eprintln!("ln: failed to access '{target_converted}': No such file or directory");
                exit_code = 1;
                continue;
            }

            match create_link(target, &link_path, kind) {
                Ok(LinkOutcome::Hardlink) | Ok(LinkOutcome::Symlink) => {
                    if verbose {
                        println!("'{}' -> '{target_converted}'", link_path.display());
                    }
                }
                Ok(LinkOutcome::Junction) => {
                    if !warned_once {
                        eprintln!("ln: symlink privilege unavailable, created junction instead");
                        warned_once = true;
                    }
                    if verbose {
                        println!("'{}' -> '{target_converted}' [junction]", link_path.display());
                    }
                }
                Err(e) => {
                    // D-38: cross-volume hard link → raw_os_error 17 (ERROR_NOT_SAME_DEVICE)
                    //        or Rust 1.85+ ErrorKind::CrossesDevices
                    let is_cross_device =
                        e.kind() == std::io::ErrorKind::CrossesDevices
                        || e.raw_os_error() == Some(17);

                    if is_cross_device && !symbolic {
                        eprintln!("ln: failed to create hard link '{}' => '{target_converted}': cross-device link not permitted",
                                  link_path.display());
                    } else {
                        eprintln!("ln: failed to create {} '{}' => '{target_converted}': {e}",
                                  if symbolic { "symbolic link" } else { "hard link" },
                                  link_path.display());
                    }
                    exit_code = 1;
                }
            }
        }
        exit_code
    }

    #[cfg(test)]
    mod tests {
        // uumain is exercised by integration tests; unit-test pure helpers only.
        // (Pattern: Shared 3 from PATTERNS.md — do not unit-test uumain.)
    }
