    ```rust
    //! `uu_mv`: GNU `mv` — move/rename files with cross-volume fallback (FILE-04).
    //!
    //! Same-volume: std::fs::rename (MoveFileExW REPLACE_EXISTING on Windows).
    //! Cross-volume: std::fs::copy + std::fs::remove_file + filetime preserve.
    //! Cross-volume directories: v1 error out (full recursive cross-vol move is v2).

    use std::ffi::OsString;
    use std::io::{self, IsTerminal, Write};
    use std::path::{Path, PathBuf};

    use clap::{Arg, ArgAction, Command};

    fn uu_app() -> Command {
        Command::new("mv")
            .about("GNU mv — move (rename) files")
            .arg(Arg::new("force").short('f').long("force").action(ArgAction::SetTrue))
            .arg(Arg::new("interactive").short('i').long("interactive").action(ArgAction::SetTrue))
            .arg(Arg::new("no-clobber").short('n').long("no-clobber").action(ArgAction::SetTrue))
            .arg(Arg::new("verbose").short('v').long("verbose").action(ArgAction::SetTrue))
            .arg(Arg::new("operands").action(ArgAction::Append).trailing_var_arg(true))
    }

    #[derive(Debug, Clone, Copy)]
    struct MvOpts {
        force: bool,
        interactive: bool,
        no_clobber: bool,
        verbose: bool,
    }

    pub fn uumain<I: IntoIterator<Item = OsString>>(args: I) -> i32 {
        gow_core::init();

        let matches = gow_core::args::parse_gnu(uu_app(), args);

        let opts = MvOpts {
            force: matches.get_flag("force"),
            interactive: matches.get_flag("interactive"),
            no_clobber: matches.get_flag("no-clobber"),
            verbose: matches.get_flag("verbose"),
        };

        let operands: Vec<String> = matches
            .get_many::<String>("operands")
            .map(|iter| iter.cloned().collect())
            .unwrap_or_default();

        if operands.len() < 2 {
            eprintln!("mv: missing file operand");
            eprintln!("usage: mv [OPTION]... SOURCE... DEST");
            return 1;
        }

        let dst_raw = operands.last().unwrap().clone();
        let sources = &operands[..operands.len() - 1];
        let dst_converted = gow_core::path::try_convert_msys_path(&dst_raw);
        let dst = Path::new(&dst_converted);
        let dst_is_dir = dst.is_dir();

        if sources.len() > 1 && !dst_is_dir {
            eprintln!("mv: target '{dst_converted}' is not a directory");
            return 1;
        }

        let mut exit_code = 0;
        for src_raw in sources {
            let src_converted = gow_core::path::try_convert_msys_path(src_raw);
            let src = Path::new(&src_converted);

            let this_dst: PathBuf = if dst_is_dir {
                dst.join(src.file_name().unwrap_or_default())
            } else {
                dst.to_path_buf()
            };

            match move_one(src, &this_dst, &opts) {
                Ok(MoveOutcome::Renamed) => {
                    if opts.verbose {
                        println!("renamed '{src_converted}' -> '{}'", this_dst.display());
                    }
                }
                Ok(MoveOutcome::Skipped) => {
                    // no-clobber or user declined prompt — silent
                }
                Err(e) => {
                    eprintln!("mv: cannot move '{src_converted}' to '{}': {e}",
                              this_dst.display());
                    exit_code = 1;
                }
            }
        }
        exit_code
    }

    enum MoveOutcome { Renamed, Skipped }

    fn move_one(src: &Path, dst: &Path, opts: &MvOpts) -> io::Result<MoveOutcome> {
        // Source existence (symlink-aware)
        let src_md = std::fs::symlink_metadata(src)?;

        // Existence checks for dst
        if dst.exists() {
            if opts.no_clobber {
                return Ok(MoveOutcome::Skipped);
            }
            if opts.interactive && !prompt_overwrite(dst) {
                return Ok(MoveOutcome::Skipped);
            }
            // -f or default: proceed (std::fs::rename on Windows replaces existing via MoveFileExW)
        }

        // Try same-volume rename first
        match std::fs::rename(src, dst) {
            Ok(()) => Ok(MoveOutcome::Renamed),
            Err(e) if is_cross_volume(&e) => {
                // Cross-volume fallback
                if src_md.is_dir() {
                    return Err(io::Error::new(
                        io::ErrorKind::CrossesDevices,
                        "cannot move directory across file system boundary (recursive copy not implemented in v1)"
                    ));
                }
                cross_volume_move(src, dst, &src_md)?;
                Ok(MoveOutcome::Renamed)
            }
            Err(e) => Err(e),
        }
    }

    fn cross_volume_move(src: &Path, dst: &Path, src_md: &std::fs::Metadata) -> io::Result<()> {
        // Capture timestamps from source first
        let atime = filetime::FileTime::from_last_access_time(src_md);
        let mtime = filetime::FileTime::from_last_modification_time(src_md);

        // Copy (std::fs::copy wraps CopyFileExW on Windows)
        std::fs::copy(src, dst)?;

        // Preserve timestamps (implicit -p behavior per RESEARCH open q1 recommendation)
        filetime::set_file_times(dst, atime, mtime)?;

        // Only after successful copy, remove source
        std::fs::remove_file(src)?;
        Ok(())
    }

    fn is_cross_volume(e: &io::Error) -> bool {
        if e.kind() == io::ErrorKind::CrossesDevices { return true; }
        #[cfg(target_os = "windows")]
        { e.raw_os_error() == Some(17) }
        #[cfg(not(target_os = "windows"))]
        { e.raw_os_error() == Some(18) }
    }

    fn prompt_overwrite(dst: &Path) -> bool {
        let mut stderr = io::stderr();
        let _ = write!(stderr, "mv: overwrite '{}'? ", dst.display());
        let _ = stderr.flush();
        // If stdin is not a tty, read whatever is piped — still allow scripted y responses.
        let _ = io::stdin().is_terminal();  // Check for information; still read below.
        let mut line = String::new();
        if io::stdin().lock().read_line(&mut line).is_err() {
            return false;
        }
        use io::BufRead as _;
        matches!(line.trim().chars().next(), Some('y' | 'Y'))
    }

    #[cfg(test)]
    mod tests {
        use super::*;

        #[test]
        fn is_cross_volume_detects_kind() {
            let e = io::Error::new(io::ErrorKind::CrossesDevices, "test");
            assert!(is_cross_volume(&e));
        }

        #[test]
        fn is_cross_volume_detects_raw_os_17() {
            #[cfg(target_os = "windows")]
            {
                let e = io::Error::from_raw_os_error(17);
                assert!(is_cross_volume(&e));
            }
        }

        #[test]
        fn is_cross_volume_negative() {
            let e = io::Error::new(io::ErrorKind::NotFound, "nope");
            assert!(!is_cross_volume(&e));
        }
    }
    ```
