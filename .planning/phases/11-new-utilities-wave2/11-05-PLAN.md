```rust
pub fn uumain<I: IntoIterator<Item = OsString>>(args: I) -> i32 {
    gow_core::init();
    let args_vec: Vec<OsString> = args.into_iter().collect();

    let mut expr_args: Vec<String> = args_vec[1..]
        .iter()
        .map(|s| s.to_string_lossy().to_string())
        .collect();

    // Detect bracket mode via --_bracket_ sentinel flag inserted by [.bat shim
    // The shim in extras/bin/[.bat calls: test.exe --_bracket_ [original args including ]
    let bracket_mode = expr_args.first().map(|s| s == "--_bracket_").unwrap_or(false);
    if bracket_mode {
        expr_args.remove(0);  // strip the --_bracket_ sentinel
        // In bracket mode, require and strip the trailing ] argument
        match expr_args.last().map(String::as_str) {
            Some("]") => { expr_args.pop(); }
            _ => {
                eprintln!("[: missing ']'");
                return 2;
            }
        }
    }

    evaluate_test(&expr_args)
}
```

```rust
fn file_test(op: &str, path: &str) -> bool {
    let converted = gow_core::path::try_convert_msys_path(path);
    match op {
        "-e" => fs::metadata(&converted).is_ok(),
        "-f" => fs::metadata(&converted).map(|m| m.is_file()).unwrap_or(false),
        "-d" => fs::metadata(&converted).map(|m| m.is_dir()).unwrap_or(false),
        "-s" => fs::metadata(&converted).map(|m| m.len() > 0).unwrap_or(false),
        "-r" => fs::metadata(&converted).is_ok(), // Windows: readable if exists
        "-w" => fs::metadata(&converted).map(|m| !m.permissions().readonly()).unwrap_or(false),

                p.extension().and_then(|e| e.to_str()),
                Some("exe") | Some("bat") | Some("com") | Some("cmd")
            )
        }
        _ => false,
    }
}
```

```rust
use std::ffi::OsString;
use std::fs;
use std::path::Path;

pub fn uumain<I: IntoIterator<Item = OsString>>(args: I) -> i32 {
    gow_core::init();
    let args_vec: Vec<OsString> = args.into_iter().collect();

    let mut expr_args: Vec<String> = args_vec[1..]
        .iter()
        .map(|s| s.to_string_lossy().to_string())
        .collect();

    // Bracket mode detection: extras/bin/[.bat calls test.exe --_bracket_ <args> ]
    let bracket_mode = expr_args.first().map(|s| s == "--_bracket_").unwrap_or(false);
    if bracket_mode {
        expr_args.remove(0);  // strip sentinel
        match expr_args.last().map(String::as_str) {
            Some("]") => { expr_args.pop(); }
            _ => {
                eprintln!("[: missing ']'");
                return 2;
            }
        }
    }

    evaluate_test(&expr_args)
}

pub fn evaluate_test(args: &[String]) -> i32 {
    if args.is_empty() {
        return 1;  // empty test → false (NOT error)
    }
    let mut pos = 0usize;
    match parse_expr(args, &mut pos) {
        Ok(result) => {
            if pos != args.len() {
                // Extra args after expression — this is a usage error in strict mode
                // but GNU test ignores trailing tokens in some cases;
                // for now: exit 2 if extra tokens remain unexpectedly
                eprintln!("test: too many arguments");
                2
            } else {
                if result { 0 } else { 1 }
            }
        }
        Err(e) => {
            eprintln!("test: {}", e);
            2
        }
    }
}
```

**Parser structure** — implement parse_expr, parse_and_expr, parse_not_expr, parse_primary:

```rust
// -o has lowest precedence, then -a, then !, then primaries
fn parse_expr(args: &[String], pos: &mut usize) -> Result<bool, String> {
    let lhs = parse_and_expr(args, pos)?;
    if *pos < args.len() && args[*pos] == "-o" {
        *pos += 1;
        let rhs = parse_expr(args, pos)?;
        return Ok(lhs || rhs);
    }
    Ok(lhs)
}

fn parse_and_expr(args: &[String], pos: &mut usize) -> Result<bool, String> {
    let lhs = parse_not_expr(args, pos)?;
    if *pos < args.len() && args[*pos] == "-a" {
        *pos += 1;
        let rhs = parse_and_expr(args, pos)?;
        return Ok(lhs && rhs);
    }
    Ok(lhs)
}

fn parse_not_expr(args: &[String], pos: &mut usize) -> Result<bool, String> {
    if *pos < args.len() && args[*pos] == "!" {
        *pos += 1;
        let inner = parse_not_expr(args, pos)?;  // ! is right-associative
        return Ok(!inner);
    }
    parse_primary(args, pos)
}
```

```rust
fn parse_primary(args: &[String], pos: &mut usize) -> Result<bool, String> {
    if *pos >= args.len() {
        return Err("missing argument".to_string());
    }

    // Parentheses
    if args[*pos] == "(" {
        *pos += 1;
        let inner = parse_expr(args, pos)?;
        if *pos >= args.len() || args[*pos] != ")" {
            return Err("missing ')'".to_string());
        }
        *pos += 1;
        return Ok(inner);
    }

    // Unary predicates (operator arg)
    let tok = &args[*pos];

    // File predicates: -f -d -e -r -w -x -s -L
    if matches!(tok.as_str(), "-f" | "-d" | "-e" | "-r" | "-w" | "-x" | "-s" | "-L") {
        let op = tok.clone();
        *pos += 1;
        if *pos >= args.len() { return Err(format!("{}: missing file argument", op)); }
        let path = &args[*pos]; *pos += 1;
        return Ok(file_test(&op, path));
    }

    // String predicates: -z and -n
    if tok == "-z" {
        *pos += 1;
        if *pos >= args.len() { return Err("-z: missing argument".to_string()); }
        let s = &args[*pos]; *pos += 1;
        return Ok(s.is_empty());
    }
    if tok == "-n" {
        *pos += 1;
        if *pos >= args.len() { return Err("-n: missing argument".to_string()); }
        let s = &args[*pos]; *pos += 1;
        return Ok(!s.is_empty());
    }

    // Binary expressions: STRING op STRING  or  INT op INT
    // These appear as 3-token sequences: arg1 op arg2
    if *pos + 2 < args.len() {
        let op = &args[*pos + 1];
        match op.as_str() {
            // String comparisons
            "=" | "!=" | "<" | ">" => {
                let a = args[*pos].clone();
                let b = args[*pos + 2].clone();
                *pos += 3;
                return Ok(match op.as_str() {
                    "=" => a == b, "!=" => a != b,
                    "<" => a < b, ">" => a > b,
                    _ => unreachable!(),
                });
            }
            // Integer comparisons
            "-eq" | "-ne" | "-lt" | "-le" | "-gt" | "-ge" => {
                let ai = args[*pos].parse::<i64>()
                    .map_err(|_| format!("integer expression expected: '{}'", args[*pos]))?;
                let bi = args[*pos + 2].parse::<i64>()
                    .map_err(|_| format!("integer expression expected: '{}'", args[*pos + 2]))?;
                *pos += 3;
                return Ok(match op.as_str() {
                    "-eq" => ai == bi, "-ne" => ai != bi,
                    "-lt" => ai < bi,  "-le" => ai <= bi,
                    "-gt" => ai > bi,  "-ge" => ai >= bi,
                    _ => unreachable!(),
                });
            }
            _ => {}
        }
    }

    // Single string argument (non-empty = true)
    let s = args[*pos].clone();
    *pos += 1;
    Ok(!s.is_empty())
}
```
