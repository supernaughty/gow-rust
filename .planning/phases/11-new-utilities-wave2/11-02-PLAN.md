fn run(cli: &Cli) -> i32 {
    let stdout = io::stdout();
    let mut out = stdout.lock();
    let mut exit_code = 0;

    if cli.files.is_empty() {
        let stdin = io::stdin();
        let reader = BufReader::new(stdin.lock());
        exit_code |= process_reader(reader, &cli, &mut out);
    } else {
        for file_path in &cli.files {
            let converted = gow_core::path::try_convert_msys_path(file_path);
            let path = Path::new(&converted);
            match File::open(path) {
                Ok(f) => {
                    let reader = BufReader::new(f);
                    exit_code |= process_reader(reader, &cli, &mut out);
                }
                Err(e) => {
                    eprintln!("<name>: {converted}: {e}");
                    exit_code = 1;
                }
            }
        }
    }
    exit_code
}

use std::ffi::OsString;
use std::fs;

pub fn uumain<I: IntoIterator<Item = OsString>>(args: I) -> i32 {
    gow_core::init();
    let args_vec: Vec<OsString> = args.into_iter().collect();
    // Skip argv[0]
    let operands: Vec<String> = args_vec[1..]
        .iter()
        .map(|s| s.to_string_lossy().to_string())
        .collect();

    if operands.len() != 1 {
        eprintln!(
            "unlink: {}",
            if operands.is_empty() {
                "missing operand".to_string()
            } else {
                format!("extra operand '{}'", operands[1])
            }
        );
        return 2;
    }

    let converted = gow_core::path::try_convert_msys_path(&operands[0]);
    match fs::remove_file(&converted) {
        Ok(()) => 0,
        Err(e) => {
            eprintln!("unlink: cannot unlink '{}': {}", converted, e);
            1
        }
    }
}

```rust
fn fmt_reader(reader: impl BufRead, width: usize, out: &mut impl Write) -> i32 {
    let mut para_words: Vec<String> = Vec::new();
    let mut exit_code = 0;

    let flush_paragraph = |para_words: &mut Vec<String>, out: &mut dyn Write| {
        if para_words.is_empty() { return; }
        // Word-wrap para_words at width, output line by line
        let mut line_len = 0usize;
        let mut first_on_line = true;
        for word in para_words.iter() {
            let word_len = word.len();
            if !first_on_line && line_len + 1 + word_len > width {
                let _ = out.write_all(b"\n");
                line_len = 0;
                first_on_line = true;
            }
            if !first_on_line {
                let _ = out.write_all(b" ");
                line_len += 1;
            }
            let _ = out.write_all(word.as_bytes());
            line_len += word_len;
            first_on_line = false;
        }
        if !first_on_line {
            let _ = out.write_all(b"\n");
        }
        para_words.clear();
    };

    for line_result in reader.lines() {
        match line_result {
            Ok(line) => {
                let trimmed = line.trim();
                if trimmed.is_empty() {
                    // Blank line = paragraph separator: flush current paragraph, then emit blank line
                    flush_paragraph(&mut para_words, out);
                    let _ = out.write_all(b"\n");
                } else {
                    // Accumulate words from this line into the paragraph buffer
                    for word in trimmed.split_whitespace() {
                        para_words.push(word.to_string());
                    }
                }
            }
            Err(e) => {
                eprintln!("fmt: read error: {e}");
                exit_code = 1;
            }
        }
    }
    // Flush final paragraph (no trailing blank line needed)
    flush_paragraph(&mut para_words, out);
    exit_code
}
```

Note: `flush_paragraph` is a closure or extracted function — either style is fine as long as it compiles. If the closure borrowing is awkward, extract it as a free function taking `para_words: &mut Vec<String>, width: usize, out: &mut impl Write`.

**crates/gow-paste/src/lib.rs** — replace stub.

Cli struct:
```rust
#[derive(Parser, Debug)]
#[command(
    name = "paste",
    about = "GNU paste — Windows port.",
    version = env!("CARGO_PKG_VERSION"),
    disable_help_flag = true,
    disable_version_flag = true
)]
struct Cli {
    #[arg(long, action = ArgAction::Help)]
    help: Option<bool>,
    #[arg(long, action = ArgAction::Version)]
    version: Option<bool>,
    /// Delimiter list for joining columns (default: tab). Cycles through if multiple.
    #[arg(short = 'd', long = "delimiters", default_value = "\t")]
    delimiters: String,
    /// Serial mode: paste one file at a time rather than in parallel columns
    #[arg(short = 's', long = "serial", action = ArgAction::SetTrue)]
    serial: bool,
    /// Files to paste (use - for stdin). If none given, reads stdin.
    files: Vec<String>,
}
```

run(cli) logic:

1. Build readers list: `Vec<Box<dyn BufRead>>`. For each operand in `cli.files`:
   - If operand == "-": push `Box::new(BufReader::new(io::stdin().lock()))`
   - Else: apply `try_convert_msys_path`, then `File::open` — on error, eprintln "paste: {path}: {err}", set exit_code=1, push a dummy reader (empty BufReader from a `Cursor::new(b"")`) so column count stays correct
   - If `cli.files.is_empty()`: treat as single "-" (stdin passthrough)

2. Parse delimiter cycle: collect `cli.delimiters` into a Vec of chars (each char is one delimiter character; the list cycles). Default is `['\t']`.

   Special escape handling in delimiter string: interpret `\t` as tab, `\n` as newline, `\\` as backslash within the delimiter string.

3. If `cli.serial` (not required for basic implementation — implement as a stub returning 0 for now): process each file line-by-line and emit all its lines joined by the delimiter, then a final newline.

4. Parallel mode (default — implement fully):
   - Loop until ALL readers are exhausted (all returned None for their line):
     - For each reader at index i:
       - Read one line (use `BufRead::lines()` — but since we need to check exhaustion per-reader, maintain a `Vec<Option<Lines<Box<dyn BufRead>>>>` OR read into a buffer)
       - Simpler approach: pre-read one line per reader into `Option<String>` per iteration
     - If ALL lines are None this iteration, break
     - Join the lines (use empty string "" for None/exhausted readers) with delimiters cycling
     - Emit joined line + "\n"
   - Delimiter cycling: `delimiters[i % delimiters.len()]` where i is the column separator index (not file index — there are N-1 delimiters for N columns)

**Key implementation note on stdin "-" repeated:** If the user writes `paste - -`, both Box<dyn BufRead> readers point to the SAME underlying stdin. When the first reader reads a line, the second reader will then read the NEXT line (they share the stream). This is correct GNU behavior — `paste - -` interleaves alternating stdin lines into two columns. No special handling needed; the sequential reads naturally alternate.

**Stdout handling:** Lock stdout once before the loop: `let stdout = io::stdout(); let mut out = stdout.lock();`

Error message format: `paste: {path}: {io_error}` (to stderr).
