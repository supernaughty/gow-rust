```rust
use bzip2::read::MultiBzDecoder;  // use Multi for real-world multi-stream .bz2 files
use bzip2::write::BzEncoder;
use bzip2::Compression;

// Compress
let mut encoder = BzEncoder::new(output_writer, Compression::default());
io::copy(&mut input_reader, &mut encoder)?;
encoder.finish()?;   // must call finish() to flush bzip2 stream

// Decompress — use MultiBzDecoder for pbzip2/Wikipedia dumps compatibility
let mut decoder = MultiBzDecoder::new(input_reader);
io::copy(&mut decoder, &mut output_writer)?;
```

## STRIDE Threat Register

| T-06-03-01 | Denial of Service | bzip2 bomb (heavily compressed data decompresses to huge output) | accept | MultiBzDecoder streams to disk/stdout; no full-memory buffer; OS disk limits apply; CLI tool not a server |
| T-06-03-02 | Tampering | output file path derivation | mitigate | Output = input + ".bz2" or strip ".bz2" — derived from input path, not from archive metadata; no traversal possible |
| T-06-03-03 | Tampering | overwrite without confirmation | accept | GNU bzip2 default behavior; matches upstream; no -i interactive mode required by R019 |
