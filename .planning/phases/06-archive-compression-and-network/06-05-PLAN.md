// Create .tar.gz (-czf)
let output_file = File::create(&archive_path)?;
let gz = GzEncoder::new(output_file, GzCompression::default());
let mut builder = Builder::new(gz);
builder.follow_symlinks(false);  // MANDATORY — GNU tar default; tar crate default is true (WRONG)
for path in &paths {
    if path.is_dir() {
        builder.append_dir_all(path.file_name().unwrap_or_default(), path)?;
    } else {
        let mut f = File::open(path)?;
        let mut header = Header::new_gnu();
        header.set_metadata(&f.metadata()?);
        header.set_path(path.file_name().unwrap_or_default())?;
        header.set_cksum();
        builder.append(&header, &mut f)?;
    }
}
let gz = builder.into_inner()?;
gz.finish()?;

// Extract .tar.gz (-xzf)
let input_file = File::open(&archive_path)?;

let gz = GzDecoder::new(input_file);

for entry in archive.entries()? {
    let entry = entry?;

}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| archive file → file system | Archive entries are written to the file system during extraction — path traversal risk |
| user-supplied paths → archive | Paths to archive come from CLI; must not escape the destination during extraction |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-05-01 | Tampering | Tar slip / path traversal in archive entries (`../` sequences) | mitigate | tar crate's `Archive::unpack()` skips entries with `..` path components by default. Do NOT call `archive.entries()` + manual `unpack_in()` bypassing the safety check. Use `archive.unpack(dest)` which has the traversal guard built in. |
| T-06-05-02 | Elevation of Privilege | Symlink extraction on Windows requires SeCreateSymbolicLinkPrivilege | accept | Log warning per Pitfall 5; continue extraction; document as known Windows limitation. Do not abort full extraction on symlink failure. |
| T-06-05-03 | Denial of Service | Archive with millions of entries consuming RAM for entry list | accept | tar crate uses streaming entries() iterator — entries are not buffered into a Vec; memory proportional to current entry header only |
| T-06-05-04 | Tampering | Overwrite existing files during extraction | accept | archive.unpack() overwrites by default (GNU tar default); matches expected behavior; no -k/--keep for tar specified in R018 |
| T-06-05-05 | Tampering | Long path names truncated in USTAR format | mitigate | Use Header::new_gnu() explicitly — GNU format uses @LongLink extension, no 100-byte truncation. Do NOT use Header::new_ustar(). |
</threat_model>
