```rust
use liblzma::read::XzDecoder;
use liblzma::write::XzEncoder;

// Compress (level 0-9; default 6 matches xz CLI default)
let mut encoder = XzEncoder::new(output_writer, 6);
io::copy(&mut input_reader, &mut encoder)?;
encoder.finish()?;   // must call finish() to flush xz stream end-of-stream marker

// Decompress
let mut decoder = XzDecoder::new(input_reader);
io::copy(&mut decoder, &mut output_writer)?;
```
