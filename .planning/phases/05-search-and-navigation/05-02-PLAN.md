#[derive(Debug, Clone, Copy, PartialEq)]
enum CmpOp { Greater, Less, Equal }

fn parse_size_spec(spec: &str) -> Result<(CmpOp, u64)> {
    let (op, rest) = match spec.as_bytes().first() {
        Some(b'+') => (CmpOp::Greater, &spec[1..]),
        Some(b'-') => (CmpOp::Less, &spec[1..]),
        _ => (CmpOp::Equal, spec),
    };
    // Trailing unit char: c (bytes, default), k (1024), M (1024^2), G (1024^3)
    let (num_str, mult): (&str, u64) = match rest.chars().last() {
        Some('c') => (&rest[..rest.len() - 1], 1),
        Some('k') => (&rest[..rest.len() - 1], 1024),
        Some('M') => (&rest[..rest.len() - 1], 1024 * 1024),
        Some('G') => (&rest[..rest.len() - 1], 1024 * 1024 * 1024),
        _ => (rest, 1),
    };
    let n: u64 = num_str.parse().map_err(|e| anyhow!("invalid -size '{}': {}", spec, e))?;
    Ok((op, n.saturating_mul(mult)))
}

fn parse_time_spec(spec: &str) -> Result<(CmpOp, u64)> {
    // identical syntax to size, but no unit suffix — value is days
    let (op, rest) = match spec.as_bytes().first() {
        Some(b'+') => (CmpOp::Greater, &spec[1..]),
        Some(b'-') => (CmpOp::Less, &spec[1..]),
        _ => (CmpOp::Equal, spec),
    };
    let n: u64 = rest.parse().map_err(|e| anyhow!("invalid time spec '{}': {}", spec, e))?;
    Ok((op, n))
}

fn cmp_size(op: CmpOp, actual: u64, target: u64) -> bool {
    match op {
        CmpOp::Greater => actual > target,
        CmpOp::Less => actual < target,
        CmpOp::Equal => actual == target,
    }
}

fn days_since(now: SystemTime, t: SystemTime) -> u64 {
    let now_s = now.duration_since(UNIX_EPOCH).map(|d| d.as_secs()).unwrap_or(0);
    let t_s = t.duration_since(UNIX_EPOCH).map(|d| d.as_secs()).unwrap_or(0);
    now_s.saturating_sub(t_s) / 86400
}

fn cmp_time(op: CmpOp, actual_days: u64, target_days: u64) -> bool {
    match op {
        CmpOp::Greater => actual_days > target_days,
        CmpOp::Less => actual_days < target_days,
        CmpOp::Equal => actual_days == target_days,
    }
}

fn exec_for_entry(cmd_parts: &[String], path: &Path) -> Result<i32> {
    if cmd_parts.is_empty() {
        return Err(anyhow!("-exec requires a command"));
    }
    let path_str = path.to_string_lossy().into_owned();
    let args: Vec<String> = cmd_parts[1..]

}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User CLI args → glob compiler | Untrusted glob pattern from argv crosses into `globset::GlobBuilder::build()`. |
| User CLI args → process exec (`-exec`) | The user-supplied command vector crosses into `std::process::Command`, which calls Win32 `CreateProcessW`. |
| Filesystem → process | `WalkDir` reads arbitrary directory metadata; symlinks/junctions could theoretically loop. |
| User-supplied paths → stdout | When `-print0` is set, raw path bytes are written to stdout; downstream consumer (e.g. xargs) receives them. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-find-01 | Tampering / Elevation of Privilege | `find -exec cmd {} \;` | mitigate | Use `std::process::Command::new(cmd).args(&args)` — NEVER `Command::new("cmd").arg("/C").arg(joined)`. Each argument is passed as a separate string to `CreateProcessW`, which preserves spaces and metacharacters verbatim. This is exactly the GOW #209 fix and the locked D-06 decision. Test `test_exec_handles_paths_with_spaces` regresses this. |
| T-05-find-02 | Denial of Service | Symlink loop in `WalkDir` | mitigate | Default `WalkDir::follow_links(false)` (D-03 implies; matches GNU `find` default without `-L`). Verified `follow_links(false)` is set in the implementation. |
| T-05-find-03 | Denial of Service | Malformed glob pattern | mitigate | `GlobBuilder::build()` returns `Err` on invalid input; we convert to `anyhow!("invalid glob '{}': {}", ...)` and exit 2. No panic, no runaway compile. |
| T-05-find-04 | Information Disclosure | Stderr leakage of path on traversal error | accept | `eprintln!("find: {e}")` may include a partial filesystem path the user technically cannot read. This matches GNU `find` behavior exactly — operational visibility outweighs the marginal risk for a CLI utility. |
| T-05-find-05 | Tampering | Null-byte injection in `-print0` stdout pipeline | mitigate | On Windows, `_setmode(1, _O_BINARY)` is called before any stdout write when `-print0` is active. This prevents CRT text-mode translation (0x1A → EOF; 0x0D 0x0A → 0x0A) from corrupting the path stream. The downstream (xargs -0) consumer relies on this. |
| T-05-find-06 | Spoofing | `-exec` running an attacker-controlled binary in the current directory | accept | If the user types `find . -exec ./evil ;`, that's exactly what GNU `find` would do. `find` does not (and should not) sanitize PATH lookups beyond what the OS does — this matches reasonable Unix expectations and is the user's responsibility. |
| T-05-find-07 | Denial of Service | Extremely deep directory tree | mitigate | `WalkDir::max_depth` enforces user-supplied or default depth limit. Without `-maxdepth`, traversal is bounded by the actual filesystem depth, which is OS-limited (Windows ~260 chars without long-path; manifest already enables long-path awareness via `embed_manifest`). |
</threat_model>
