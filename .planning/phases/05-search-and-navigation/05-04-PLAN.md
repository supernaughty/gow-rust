    loop {

                    match (code, modifiers) {
                        (KeyCode::Char('q'), _) | (KeyCode::Char('c'), KeyModifiers::CONTROL) => break,
                        (KeyCode::Up, _) | (KeyCode::Char('k'), _) => state.scroll_up(1),

                        (KeyCode::PageUp, _) | (KeyCode::Char('b'), _) => state.scroll_up(state.viewport_h),

                        (KeyCode::Char('/'), _) => state.enter_search()?,
                        (KeyCode::Char('n'), _) => state.next_match(),
                        (KeyCode::Char('N'), _) => state.prev_match(),
                        _ => {}
                    }
                }
                Event::Resize(w, h) => state.resize(w, h),
                _ => {}
            }
            state.render()?;
        }
    }
    Ok(())
}

//! Lazy line-offset index for the `less` pager.
//!
//! Stores `Vec<u64>` of byte offsets, one per line start. Builds incrementally
//! as the user scrolls forward. Never loads the full file content into memory
//! — only the offset table grows (~8 bytes per line).
//!
//! Implements decision D-09: streaming I/O, no full-file buffering.

use std::fs::File;
use std::io::{self, BufRead, BufReader, Seek, SeekFrom};

/// Lazy line-offset index over a seekable file.
pub struct LineIndex {
    /// Byte offsets where each line starts. `offsets[i]` is the start of line `i`.
    /// After scanning past EOF, the final entry is the file's total length.
    offsets: Vec<u64>,
    reader: BufReader<File>,
    eof_reached: bool,
}

impl LineIndex {
    pub fn new(file: File) -> Self {
        Self {
            offsets: vec![0],
            reader: BufReader::new(file),
            eof_reached: false,
        }
    }

    /// Read forward until `line_num` is indexed or EOF.
    pub fn ensure_indexed_to(&mut self, line_num: usize) -> io::Result<()> {
        // Position reader at the last known offset before scanning.
        if !self.eof_reached && self.offsets.len() <= line_num {
            let last_known = *self.offsets.last().unwrap();
            self.reader.seek(SeekFrom::Start(last_known))?;
        }
        while self.offsets.len() <= line_num && !self.eof_reached {
            let mut buf = Vec::new();
            let n = self.reader.read_until(b'\n', &mut buf)?;
            if n == 0 {
                self.eof_reached = true;
                break;
            }
            let next_offset = self.offsets.last().unwrap() + n as u64;
            self.offsets.push(next_offset);
        }
        Ok(())
    }

    /// Scan to EOF, returning total line count.
    pub fn scan_to_end(&mut self) -> io::Result<usize> {
        // Position reader at the last indexed offset.
        if !self.eof_reached {
            let last_known = *self.offsets.last().unwrap();
            self.reader.seek(SeekFrom::Start(last_known))?;
        }
        while !self.eof_reached {
            let mut buf = Vec::new();
            let n = self.reader.read_until(b'\n', &mut buf)?;
            if n == 0 {
                self.eof_reached = true;
                break;
            }
            let next_offset = self.offsets.last().unwrap() + n as u64;
            self.offsets.push(next_offset);
        }
        Ok(self.line_count_so_far())
    }

    /// Number of lines currently indexed (not counting the trailing EOF marker).
    pub fn line_count_so_far(&self) -> usize {
        // offsets always contains [0, ..., end]. If only [0] and !eof, 0 lines indexed.
        // If [0, len] and eof, that means an empty file or content with no newline → 1 line.
        if self.offsets.len() <= 1 {
            if self.eof_reached { 0 } else { 0 }
        } else {
            self.offsets.len() - 1
        }
    }

    pub fn is_complete(&self) -> bool {
        self.eof_reached
    }

    /// Read line `line_num` (0-based). Returns `Ok(None)` if not yet indexed/past EOF.
    /// The returned bytes include any trailing `\n` if present in the file.
    pub fn read_line_at(&mut self, line_num: usize) -> io::Result<Option<Vec<u8>>> {
        self.ensure_indexed_to(line_num + 1)?; // need offset of line_num+1 to know length
        if line_num >= self.line_count_so_far() {
            return Ok(None);
        }
        let start = self.offsets[line_num];
        self.reader.seek(SeekFrom::Start(start))?;
        let mut buf = Vec::new();
        self.reader.read_until(b'\n', &mut buf)?;
        Ok(Some(buf))
    }
}

#[derive(Parser, Debug)]
#[command(
    name = "less",
    about = "GNU less — Windows port.",
    version = env!("CARGO_PKG_VERSION"),
    disable_help_flag = true,
    disable_version_flag = true,
)]
struct Cli {
    /// File to page (default: stdin)
    file: Option<PathBuf>,
}

pub fn uumain<I: IntoIterator<Item = OsString>>(args: I) -> i32 {
    gow_core::init();
    let matches = gow_core::args::parse_gnu(Cli::command(), args);
    let cli = match Cli::from_arg_matches(&matches) {
        Ok(c) => c,
        Err(e) => { eprintln!("less: {}", e); return 2; }
    };

    // Non-TTY fallback: behave like `cat`. Critical for CI testing.
    if !io::stdout().is_tty() {
        return match copy_to_stdout(&cli) {
            Ok(()) => 0,
            Err(e) => { eprintln!("less: {}", e); 1 }
        };
    }

    match run_pager(cli) {
        Ok(()) => 0,
        Err(e) => { eprintln!("less: {}", e); 1 }
    }
}

/// Non-interactive mode: stream source to stdout byte-for-byte.
fn copy_to_stdout(cli: &Cli) -> Result<()> {
    let mut out = io::stdout().lock();
    match &cli.file {
        Some(path) => {
            let mut f = File::open(path)
                .map_err(|e| anyhow!("{}: {}", path.display(), e))?;
            io::copy(&mut f, &mut out)?;
        }
        None => {
            let stdin = io::stdin();
            let mut s = stdin.lock();
            io::copy(&mut s, &mut out)?;
        }
    }
    out.flush()?;
    Ok(())
}

/// Open the source as a seekable file. For stdin, buffer to a temp file.
fn open_source(cli: &Cli) -> Result<LineIndex> {
    match &cli.file {
        Some(path) => {
            let f = File::open(path).map_err(|e| anyhow!("{}: {}", path.display(), e))?;
            Ok(LineIndex::new(f))
        }
        None => {
            // Stdin is non-seekable; buffer to a temp file (matches GNU less behavior).
            // Acceptable per RESEARCH.md A3.
            let mut tmp = tempfile::NamedTempFile::new()?;
            io::copy(&mut io::stdin().lock(), tmp.as_file_mut())?;
            tmp.as_file().sync_all()?;
            let f = File::open(tmp.path())?;
            Ok(LineIndex::new(f))
        }
    }
}

struct TerminalGuard;

impl Drop for TerminalGuard {
    fn drop(&mut self) {
        let _ = io::stdout().execute(LeaveAlternateScreen);
        let _ = disable_raw_mode();
    }
}

fn run_pager(cli: Cli) -> Result<()> {
    // Install panic hook BEFORE enable_raw_mode (Pitfall 1 / Threat T-05-less-01).
    let default_hook = panic::take_hook();
    panic::set_hook(Box::new(move |info| {
        let _ = io::stdout().execute(LeaveAlternateScreen);
        let _ = disable_raw_mode();
        default_hook(info);
    }));

    enable_raw_mode()?;
    io::stdout().execute(EnterAlternateScreen)?;
    let _guard = TerminalGuard; // restored on every exit path (normal, ?, panic).

    let mut state = PagerState::new(open_source(&cli)?)?;
    state.render()?;
    event_loop(&mut state)?;
    Ok(())
}

struct PagerState {
    index: LineIndex,
    top_line: usize,
    viewport_h: u16,
    viewport_w: u16,
    search_pattern: Option<Regex>,
    match_lines: Vec<usize>,
    current_match_idx: Option<usize>,
}

impl PagerState {
    fn new(index: LineIndex) -> Result<Self> {
        let (cols, rows) = match terminal_size() {
            Some((Width(w), Height(h))) => (w, h),
            None => (80, 24),
        };
        Ok(Self {
            index,
            top_line: 0,
            viewport_h: rows,
            viewport_w: cols,
            search_pattern: None,
            match_lines: Vec::new(),
            current_match_idx: None,
        })
    }

    fn body_height(&self) -> usize {
        // Reserve 1 line for the status row.
        self.viewport_h.saturating_sub(1) as usize
    }

    fn render(&mut self) -> Result<()> {
        let mut out = io::stdout().lock();
        out.execute(Clear(ClearType::All))?;
        out.execute(MoveTo(0, 0))?;
        out.execute(ResetColor)?;
        let h = self.body_height();
        for i in 0..h {
            let line_num = self.top_line + i;
            self.index.ensure_indexed_to(line_num + 1)?;
            if let Some(line) = self.index.read_line_at(line_num)? {
                // Write raw bytes — preserves ANSI escape sequences (D-08).
                out.write_all(&line)?;
                if !line.ends_with(b"\n") {
                    out.write_all(b"\n")?;
                }
            } else {
                out.write_all(b"~\n")?; // GNU less convention for past-EOF lines
            }
        }
        // Status line
        out.execute(MoveTo(0, self.viewport_h.saturating_sub(1)))?;
        let prompt = match &self.search_pattern {
            Some(p) => format!(":/{}/ (n=next, N=prev, q=quit) ", p.as_str()),
            None => ":".to_string(),
        };
        out.write_all(prompt.as_bytes())?;
        out.flush()?;
        Ok(())
    }

    fn scroll_up(&mut self, n: usize) {
        self.top_line = self.top_line.saturating_sub(n);
    }

    fn scroll_down(&mut self, n: usize) {
        // Allow scrolling forward even if EOF unknown; ensure_indexed_to in render will fetch.
        self.top_line = self.top_line.saturating_add(n);
    }

    fn jump_to_start(&mut self) {
        self.top_line = 0;
    }

    fn jump_to_end(&mut self) -> Result<()> {
        let total = self.index.scan_to_end()?;
        self.top_line = total.saturating_sub(self.body_height());
        Ok(())
    }

    fn enter_search(&mut self) -> Result<()> {
        // Read keystrokes until Enter to build pattern.
        let mut pattern = String::new();
        loop {
            let mut out = io::stdout().lock();
            out.execute(MoveTo(0, self.viewport_h.saturating_sub(1)))?;
            out.execute(Clear(ClearType::CurrentLine))?;
            write!(out, "/{}", pattern)?;
            out.flush()?;
            drop(out);

            if let Event::Key(KeyEvent { code, modifiers, .. }) = event::read()? {
                match (code, modifiers) {
                    (KeyCode::Enter, _) => break,
                    (KeyCode::Esc, _) => return Ok(()),
                    (KeyCode::Char('c'), KeyModifiers::CONTROL) => return Ok(()),
                    (KeyCode::Backspace, _) => { pattern.pop(); }
                    (KeyCode::Char(c), _) => pattern.push(c),
                    _ => {}
                }
            }
        }

        if pattern.is_empty() { return Ok(()); }

        let re = Regex::new(&pattern).map_err(|e| anyhow!("bad regex: {}", e))?;
        self.search_pattern = Some(re);
        self.match_lines.clear();
        self.current_match_idx = None;
        // Pre-compute matches in already-indexed lines and extend forward until first match.
        self.find_initial_matches()?;
        if let Some(&first) = self.match_lines.first() {
            self.top_line = first;
            self.current_match_idx = Some(0);
        }
        Ok(())
    }

    fn find_initial_matches(&mut self) -> Result<()> {
        let re = match &self.search_pattern { Some(r) => r.clone(), None => return Ok(()) };
        // Scan from line 0 onward; extend index up to the first match (or EOF).
        let mut line = 0usize;
        loop {
            self.index.ensure_indexed_to(line + 1)?;
            if line >= self.index.line_count_so_far() && self.index.is_complete() {
                break;
            }
            if let Some(buf) = self.index.read_line_at(line)? {
                if re.is_match(std::str::from_utf8(&buf).unwrap_or("")) {
                    self.match_lines.push(line);
                }
            } else {
                break;
            }
            line += 1;
            if line > 100_000 { break; } // safety cap; gap-closure can lift
        }
        Ok(())
    }

    fn next_match(&mut self) {
        if self.match_lines.is_empty() { return; }
        let next = match self.current_match_idx {
            Some(i) if i + 1 < self.match_lines.len() => i + 1,
            _ => 0,
        };
        self.current_match_idx = Some(next);
        self.top_line = self.match_lines[next];
    }

    fn prev_match(&mut self) {
        if self.match_lines.is_empty() { return; }
        let prev = match self.current_match_idx {
            Some(i) if i > 0 => i - 1,
            _ => self.match_lines.len() - 1,
        };
        self.current_match_idx = Some(prev);
        self.top_line = self.match_lines[prev];
    }

    fn resize(&mut self, w: u16, h: u16) {
        self.viewport_w = w;
        self.viewport_h = h;
    }
}

fn event_loop(state: &mut PagerState) -> Result<()> {
    loop {
        if event::poll(Duration::from_millis(250))? {
            match event::read()? {
                Event::Key(KeyEvent { code, modifiers, .. }) => {
                    match (code, modifiers) {
                        (KeyCode::Char('q'), _)
                        | (KeyCode::Char('c'), KeyModifiers::CONTROL) => break,
                        (KeyCode::Up, _) | (KeyCode::Char('k'), _) => state.scroll_up(1),
                        (KeyCode::Down, _) | (KeyCode::Char('j'), _) => state.scroll_down(1),
                        (KeyCode::PageUp, _) | (KeyCode::Char('b'), _) => {
                            state.scroll_up(state.body_height());
                        }
                        (KeyCode::PageDown, _) | (KeyCode::Char(' '), _) => {
                            state.scroll_down(state.body_height());
                        }
                        (KeyCode::Char('g'), _) => state.jump_to_start(),
                        (KeyCode::Char('G'), _) => state.jump_to_end()?,
                        (KeyCode::Char('/'), _) => state.enter_search()?,
                        (KeyCode::Char('n'), _) => state.next_match(),
                        (KeyCode::Char('N'), _) => state.prev_match(),
                        _ => {}
                    }
                }
                Event::Resize(w, h) => state.resize(w, h),
                _ => {}
            }
            state.render()?;
        }
    }
    Ok(())
}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User CLI args → file open | A user-supplied path is opened with `File::open`. |
| File / stdin → in-memory line index | Untrusted bytes (file contents) are read into the LineIndex; offsets are derived from the byte stream. |
| stdin (non-seekable) → temp file | When piped, stdin is buffered to a `tempfile::NamedTempFile` so it can be seeked. |
| Process state → terminal | Raw mode + alternate screen put the user's terminal into a fragile state; restoration is required on every exit path. |
| Untrusted regex pattern → regex compiler | The user's `/pattern` keystrokes go to `Regex::new`. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-less-01 | Denial of Service / User Impact | Terminal corruption on panic (RESEARCH.md Pitfall 1) | mitigate | `panic::set_hook` is installed BEFORE `enable_raw_mode()` and calls `LeaveAlternateScreen` + `disable_raw_mode` before delegating to the default hook. Additionally, `TerminalGuard` with `impl Drop` runs on every normal-exit path (including `?`-propagated errors). Both are required because the hook handles `panic!()` while the guard handles graceful exit. Acceptance criteria assert source-order: hook installed before `enable_raw_mode`. |
| T-05-less-02 | Denial of Service | Memory exhaustion on large files (RESEARCH.md Pitfall, D-09) | mitigate | `LineIndex` stores only `Vec<u64>` offsets (~8 bytes/line). Line content is read on demand via `seek` + `read_until`. Test `test_less_large_file_streams_without_oom` regresses this with 1 MiB. The known limitation: `G` (jump to end) blocks the event loop while scanning; documented and accepted per RESEARCH.md Pitfall 5 / A2. |
| T-05-less-03 | Denial of Service | Stdin buffering OOM | accept | Per D-09 + RESEARCH.md A3, when stdin is piped (non-seekable), we buffer to a temp file. This is bounded by available disk, not RAM — matches GNU `less` behavior. Acceptable tradeoff. |
| T-05-less-04 | Denial of Service | Pathological regex (`(a+)+b`-style) in `/search` | accept | We use the `regex` crate (linear-time guarantees, no catastrophic backtracking — verified in CLAUDE.md tech stack). Pathological patterns cannot blow up. The user typing `/` is local interaction; not an attack surface in any realistic scenario. |
| T-05-less-05 | Information Disclosure | ANSI escape sequence injection | accept | We deliberately pass ANSI bytes through unchanged (D-08). A file containing terminal control sequences could move the cursor or clear the screen. This is exactly what `less -R` does and what GOW users expect from `grep --color | less`. The alternate screen confines damage; the RAII guard cleans up on exit. Standard pager behavior; not a vulnerability. |
