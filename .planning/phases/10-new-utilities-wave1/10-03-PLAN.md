pub fn uumain<I: IntoIterator<Item = OsString>>(args: I) -> i32 {
    gow_core::init();
    let args_vec: Vec<OsString> = args.into_iter().collect();

    // Detect invocation name for argv[0] mode switching
    let invoked_as = args_vec
        .first()
        .map(|s| {
            std::path::Path::new(s)
                .file_stem()
                .unwrap_or_default()
                .to_string_lossy()
                .to_lowercase()
        })
        .unwrap_or_default();

    let matches = gow_core::args::parse_gnu(Cli::command(), args_vec);
    let cli = match Cli::from_arg_matches(&matches) {
        Ok(c) => c,
        Err(e) => { eprintln!("expand: {e}"); return 2; }
    };

    run(&cli, &invoked_as)
}
```

          version = env!("CARGO_PKG_VERSION"),
          disable_help_flag = true, disable_version_flag = true)]
struct Cli {
    #[arg(long, action = ArgAction::Help)]
    help: Option<bool>,
    #[arg(long, action = ArgAction::Version)]
    version: Option<bool>,
    /// Body numbering: a (all), t (non-empty, default), n (none)
    #[arg(short = 'b', long = "body-numbering", default_value = "t")]
    body_numbering: String,
    /// Line number field width (default 6)
    #[arg(short = 'w', long = "number-width", default_value = "6")]
    width: usize,
    /// Separator between number and line (default tab)
    #[arg(short = 's', long = "number-separator", default_value = "\t")]
    separator: String,
    /// Starting line number (default 1)
    #[arg(short = 'v', long = "starting-line-number", default_value = "1")]
    start: i64,
    /// Increment between line numbers (default 1)
    #[arg(short = 'i', long = "line-increment", default_value = "1")]
    increment: i64,
    files: Vec<String>,
}
```

run(cli) logic:
1. Validate body_numbering is one of "a", "t", "n" — else eprintln "nl: invalid body numbering style: '<x>'", return 1
2. If files.is_empty(): process stdin
3. Else for each file: open, process, on error eprintln "nl: <path>: <err>", set exit_code=1
4. process(reader):
   - line_num = cli.start
   - For each line (use BufRead::lines or read whole then split):
     - If body_numbering == "n": emit `<separator><line>\n`
     - If body_numbering == "t" AND line is empty (zero-length after stripping \r): emit just `\n` (no number, no padding, no separator)
     - If body_numbering == "t" with non-empty line OR body_numbering == "a": emit `{:>width$}<separator><line>\n`, then `line_num += increment`

Format the number with right-alignment in `width` chars: `format!("{:>width$}", line_num, width=cli.width)`.

**gow-expand-unexpand/src/lib.rs** — replace stub. This is a SHARED implementation; argv[0] selects the mode.

```rust
use std::ffi::OsString;
use std::fs::File;
use std::io::{self, BufRead, BufReader, Read, Write};
use std::path::Path;

use clap::{ArgAction, CommandFactory, FromArgMatches, Parser};

#[derive(Parser, Debug)]
#[command(name = "expand", about = "GNU expand/unexpand — Windows port.",
          version = env!("CARGO_PKG_VERSION"),
          disable_help_flag = true, disable_version_flag = true)]
struct Cli {
    #[arg(long, action = ArgAction::Help)]
    help: Option<bool>,
    #[arg(long, action = ArgAction::Version)]
    version: Option<bool>,
    /// Tab stop width (default 8)
    #[arg(short = 't', long = "tabs", default_value = "8")]
    tabs: usize,
    /// (unexpand only) Convert all blanks, not just leading
    #[arg(short = 'a', long = "all", action = ArgAction::SetTrue)]
    all: bool,
    files: Vec<String>,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
enum Mode { Expand, Unexpand }

fn detect_mode(invoked_as: &str) -> Mode {
    if invoked_as == "unexpand" { Mode::Unexpand } else { Mode::Expand }
}

fn expand_line(line: &str, tab_width: usize) -> String {
    let mut out = String::with_capacity(line.len());
    let mut col = 0usize;
    for ch in line.chars() {
        if ch == '\t' {
            let pad = tab_width - (col % tab_width);
            for _ in 0..pad { out.push(' '); }
            col += pad;
        } else {
            out.push(ch);
            // For \n/\r, reset col would be more accurate but expand operates one
            // line at a time so this is acceptable. Phase 10 simplification.
            if ch == '\n' { col = 0; } else { col += 1; }
        }
    }
    out
}

fn unexpand_line(line: &str, tab_width: usize, all_blanks: bool) -> String {
    // GNU unexpand default: only convert LEADING whitespace into tabs.
    // -a: convert all whitespace runs of >=2 spaces or tab-equivalents.
    let mut out = String::with_capacity(line.len());
    let mut chars: Vec<char> = line.chars().collect();
    let mut i = 0;
    let mut in_leading = true;
    while i < chars.len() {
        // Process a run of spaces starting at i, only if (in_leading) || all_blanks
        if chars[i] == ' ' && (in_leading || all_blanks) {
            // Count consecutive spaces
            let start = i;
            while i < chars.len() && chars[i] == ' ' { i += 1; }
            let count = i - start;
            // Determine column at `start` based on what we have emitted so far (use out length)
            // Since out is ASCII for the leading run, byte-len approximates col.
            let col = out.chars().count();
            // Compute next tab stop from `col`
            let to_next = tab_width - (col % tab_width);
            if count >= to_next {
                // Emit tabs first to align
                let mut remaining = count;
                let mut cur_col = col;
                // First tab
                if remaining >= to_next {
                    out.push('\t');
                    remaining -= to_next;
                    cur_col += to_next;
                }
                // Subsequent full-width tabs
                while remaining >= tab_width {
                    out.push('\t');
                    remaining -= tab_width;
                    cur_col += tab_width;
                }
                // Trailing residual spaces
                for _ in 0..remaining { out.push(' '); }
                let _ = cur_col;
            } else {
                // Not enough spaces to align — keep as-is
                for _ in 0..count { out.push(' '); }
            }
            continue;
        }
        let c = chars[i];
        if c != ' ' && c != '\t' { in_leading = false; }
        if c == '\n' || c == '\r' { in_leading = true; } // reset for next line
        out.push(c);
        i += 1;
    }
    out
}

fn process<R: Read, W: Write>(mode: Mode, cli: &Cli, reader: R, mut writer: W) -> io::Result<()> {
    let buf = BufReader::new(reader);
    for line_res in buf.lines() {
        let line = line_res?;
        let transformed = match mode {
            Mode::Expand => expand_line(&format!("{line}\n"), cli.tabs),
            Mode::Unexpand => unexpand_line(&format!("{line}\n"), cli.tabs, cli.all),
        };
        writer.write_all(transformed.as_bytes())?;
    }
    Ok(())
}

fn run(cli: &Cli, invoked_as: &str) -> i32 {
    let mode = detect_mode(invoked_as);
    if cli.tabs == 0 {
        eprintln!("{}: tab size must be positive", invoked_as);
        return 1;
    }
    let mut exit_code = 0;
    if cli.files.is_empty() {
        let stdin = io::stdin();
        let stdout = io::stdout();
        if let Err(e) = process(mode, cli, stdin.lock(), stdout.lock()) {
            eprintln!("{}: stdin: {e}", invoked_as);
            return 1;
        }
        return 0;
    }
    for f in &cli.files {
        let converted = gow_core::path::try_convert_msys_path(f);
        match File::open(Path::new(&converted)) {
            Ok(file) => {
                let stdout = io::stdout();
                if let Err(e) = process(mode, cli, file, stdout.lock()) {
                    eprintln!("{}: {converted}: {e}", invoked_as);
                    exit_code = 1;
                }
            }
            Err(e) => {
                eprintln!("{}: {converted}: {e}", invoked_as);
                exit_code = 1;
            }
        }
    }
    exit_code
}

pub fn uumain<I: IntoIterator<Item = OsString>>(args: I) -> i32 {
    gow_core::init();
    let args_vec: Vec<OsString> = args.into_iter().collect();
    let invoked_as = args_vec
        .first()
        .map(|s| Path::new(s).file_stem().unwrap_or_default().to_string_lossy().to_lowercase())
        .unwrap_or_default();
    let matches = gow_core::args::parse_gnu(Cli::command(), args_vec);
    let cli = match Cli::from_arg_matches(&matches) {
        Ok(c) => c,
        Err(e) => { eprintln!("{}: {e}", if invoked_as == "unexpand" { "unexpand" } else { "expand" }); return 2; }
    };
    run(&cli, &invoked_as)
}
```
