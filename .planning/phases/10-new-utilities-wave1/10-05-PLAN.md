```rust
fn human_readable(bytes: u64) -> String {
    const UNITS: &[(&str, u64)] = &[
        ("E", 1 << 60), ("P", 1 << 50), ("T", 1 << 40),
        ("G", 1 << 30), ("M", 1 << 20), ("K", 1 << 10),
    ];
    for (suffix, factor) in UNITS {
        if bytes >= *factor {
            let val = bytes as f64 / *factor as f64;
            return if val < 10.0 {
                format!("{:.1}{}", val, suffix)
            } else {
                format!("{:.0}{}", val, suffix)
            };
        }
    }
    format!("{}", bytes)
}
```

```rust
use std::os::windows::ffi::OsStringExt;
use windows_sys::Win32::Storage::FileSystem::{GetDiskFreeSpaceExW, GetLogicalDriveStringsW};

fn get_drives() -> Vec<String> {
    let mut buf = [0u16; 256];
    let len = unsafe { GetLogicalDriveStringsW(buf.len() as u32, buf.as_mut_ptr()) };
    let mut drives = Vec::new();
    let mut start = 0usize;
    for i in 0..len as usize {
        if buf[i] == 0 {
            if i > start {
                drives.push(String::from_utf16_lossy(&buf[start..i]));
            }
            start = i + 1;
        }
    }
    drives
}

unsafe fn get_disk_free(root: &str) -> Option<(u64, u64, u64)> {
    let wide: Vec<u16> = root.encode_utf16().chain(std::iter::once(0)).collect();
    let mut free_available: u64 = 0;
    let mut total_bytes: u64 = 0;
    let mut total_free: u64 = 0;
    let ok = GetDiskFreeSpaceExW(
        wide.as_ptr(),
        &mut free_available,
        &mut total_bytes,
        &mut total_free,
    );
    if ok != 0 { Some((free_available, total_bytes, total_free)) } else { None }
}
```

```rust
use walkdir::WalkDir;

fn dir_usage(path: &std::path::Path, max_depth: Option<usize>) -> u64 {
    let mut walker = WalkDir::new(path).follow_links(false);
    if let Some(d) = max_depth { walker = walker.max_depth(d); }
    walker.into_iter()
        .filter_map(|e| e.ok())
        .filter(|e| e.file_type().is_file())
        .map(|e| e.metadata().map(|m| m.len()).unwrap_or(0))
        .sum()
}
```

`metadata().len()` returns the size in bytes. For du output:
- Default block size = 1024 bytes (1K blocks). Output column = `(bytes + 1023) / 1024` (round up like GNU).
- With -h: format with human_readable().

GNU df default columns (with single space alignment, NOT tab-separated):

```
Filesystem     1K-blocks      Used Available Use% Mounted on
C:\           487654321 234567890 253086431  49% C:\
```

GNU df -h columns:

```
Filesystem      Size  Used Avail Use% Mounted on
C:\             465G  224G  242G  49% C:\
```

For Windows: "Filesystem" and "Mounted on" are both the drive letter root (e.g. "C:\"). Used = total_bytes - free_available; Use% = (used * 100 + total_bytes/2) / total_bytes (round half up).

run(cli) logic:
1. If `cli.summarize && cli.max_depth.is_some()`: error "du: cannot both summarize and show all entries", return 1
2. Determine paths: cli.paths if non-empty, else `["."]`
3. For each path:
   - Convert via gow_core::path::try_convert_msys_path
   - If it doesn't exist: eprintln "du: cannot access '<p>': No such file or directory", set exit_code=1, continue
   - If summarize: compute total via walkdir, emit one line with `format_size(total) + "\t" + display_path + "\n"`
   - Else: walk and collect per-directory totals (post-order: total = sum of children + own files); emit a line for each directory (and -a: each file too) up to max_depth depth
4. format_size(bytes):
   - If cli.human: use the human_readable() helper from interfaces (RESEARCH Pattern 5)
   - Else if cli.bytes: format bytes as-is
   - Else (default): print 1K blocks: `format!("{}", (bytes + 1023) / 1024)`
5. Walkdir setup: `WalkDir::new(p).follow_links(false)` (CRITICAL — see Pitfall 3); apply `.max_depth(d+1)` for cli.max_depth (off by one because depth 0 = the root entry itself)

For per-directory totals (default non-summarize mode), use the post-order pattern:
- WalkDir gives entries in pre-order with `.contents_first(false)` (default); switching to `.contents_first(true)` gives post-order which is what we need
- Maintain a HashMap<PathBuf, u64> of directory totals
- When iterating entries:
  - For files: add metadata().len() to the total of EACH ancestor up to and including the current root path (or just the immediate parent — simpler: aggregate by walking)
- Simpler approach: for each directory at depth <= max_depth, run a separate WalkDir summing file sizes within. O(N²) but acceptable for Phase 10.

For correctness AND simplicity, use this approach:

```rust
fn dir_usage_recursive(path: &Path) -> u64 {
    WalkDir::new(path).follow_links(false).into_iter()
        .filter_map(|e| e.ok())
        .filter(|e| e.file_type().is_file())
        .map(|e| e.metadata().map(|m| m.len()).unwrap_or(0))
        .sum()
}

fn run_one(cli: &Cli, root: &Path, exit: &mut i32) {
    if !root.exists() {
        eprintln!("du: cannot access '{}': No such file or directory", root.display());
        *exit = 1;
        return;
    }
    if cli.summarize {
        let total = dir_usage_recursive(root);
        println!("{}\t{}", format_size(total, cli), root.display());
        return;
    }
    // Non-summarize: emit per-directory totals up to max_depth
    let max_depth = cli.max_depth.unwrap_or(usize::MAX);
    for entry in WalkDir::new(root).follow_links(false).max_depth(max_depth) {
        match entry {
            Ok(e) => {
                let is_dir = e.file_type().is_dir();
                let is_file = e.file_type().is_file();
                if is_dir || (cli.all && is_file) {
                    let total = if is_dir {
                        dir_usage_recursive(e.path())
                    } else {
                        e.metadata().map(|m| m.len()).unwrap_or(0)
                    };
                    println!("{}\t{}", format_size(total, cli), e.path().display());
                }
            }
            Err(err) => {
                eprintln!("du: {err}");
                *exit = 1;
            }
        }
    }
}
```

(Note: `dir_usage_recursive` is called once per directory in the second loop, which is O(N²). Acceptable for Phase 10. A linear post-order accumulator is a stretch goal.)

```rust
unsafe fn get_disk_free(root: &str) -> Option<(u64, u64, u64)> {

fn human_readable(bytes: u64) -> String {
    const UNITS: &[(&str, u64)] = &[
        ("E", 1u64 << 60), ("P", 1u64 << 50), ("T", 1u64 << 40),
        ("G", 1u64 << 30), ("M", 1u64 << 20), ("K", 1u64 << 10),

    for (suffix, factor) in UNITS {

            return if val < 10.0 {

            } else {
                format!("{:.0}{}", val, suffix)
            };
        }
    }
    format!("{}", bytes)
}

fn format_size(bytes: u64, human: bool) -> String {
    if human { human_readable(bytes) }
    else { format!("{}", (bytes + 1023) / 1024) }
}

fn percent_used(total: u64, free: u64) -> String {
    if total == 0 { return "-".to_string(); }
    let used = total.saturating_sub(free);
    let pct = ((used as f64 / total as f64) * 100.0).round() as u64;
    format!("{}%", pct)
}
```

```rust
fn run(cli: &Cli) -> i32 {
    let drives = get_drives();
    if drives.is_empty() {
        eprintln!("df: no drives detected");
        return 1;
    }

    // Header
    let stdout = std::io::stdout();
    let mut out = stdout.lock();
    if cli.human {
        let _ = writeln!(out, "Filesystem      Size  Used Avail Use% Mounted on");
    } else {
        let _ = writeln!(out, "Filesystem     1K-blocks      Used Available Use% Mounted on");
    }

    for drive in &drives {
        let info = unsafe { get_disk_free(drive) };
        let Some((free_available, total_bytes, _total_free)) = info else {
            // Pitfall 4: skip drives that respond FALSE (CD-ROM with no media, etc.)
            continue;
        };
        let used = total_bytes.saturating_sub(free_available);
        let size_col = format_size(total_bytes, cli.human);
        let used_col = format_size(used, cli.human);
        let avail_col = format_size(free_available, cli.human);
        let pct_col = percent_used(total_bytes, free_available);
        // Right-pad/right-align columns; widths chosen to mirror GNU df
        if cli.human {
            let _ = writeln!(out, "{:<14} {:>5} {:>5} {:>5} {:>4} {}",
                drive, size_col, used_col, avail_col, pct_col, drive);
        } else {
            let _ = writeln!(out, "{:<14} {:>9} {:>9} {:>9} {:>4} {}",
                drive, size_col, used_col, avail_col, pct_col, drive);
        }
    }
    0
}

pub fn uumain<I: IntoIterator<Item = OsString>>(args: I) -> i32 {

        Ok(c) => c,
        Err(e) => { eprintln!("df: {e}"); return 2; }

}
```
